**코드(선택)**:

```cpp

#include <string>
#include <vector>

using namespace std;

vector<int> solution(vector<int> arr, int n) {
    vector<int> answer;
    
    for (int i = 0; i < arr.size(); i++)
    {
        int tmp = arr[i];
        if (arr.size() % 2 == 1)
        {
            if (i % 2 == 0) 
                answer.push_back(tmp + n);
            else
                answer.push_back(tmp);
        }
        
        else
        {
            if (i % 2 == 1) 
                answer.push_back(tmp + n);
            else
                answer.push_back(tmp);
        }
    }
    return answer;
}
```
