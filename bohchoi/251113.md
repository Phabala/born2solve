#include <string>
#include <vector>

using namespace std;

vector<int> solution(vector<int> arr, vector<vector<int>> queries) {
    vector<int> answer;
    int len = queries.size();
    int start, end;
    for (int i = 0; i < len; i++)
    {
        start = queries[i].at(0);
        end = queries[i].at(1);
        for (int j = start; j <= end; j++){
            arr.at(j) += 1;
        }
    }
    answer.swap(arr);
    return answer;
}
