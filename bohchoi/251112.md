#include <string>
#include <vector>

using namespace std;

int check_digit(int num)
{
    int i = 1;
    
    while (num >= 10)
    {
        num /= 10;
        i++;
    }
    return (i);
}

int up_digit(int num, int digit)
{
    for (int i = 0; i < check_digit(digit); i++)
    {
        num *= 10;
    }
    return (num + digit);
}

int solution(int a, int b) {
    int answer = 0;
    int num1 = up_digit(a, b);
    int num2 = up_digit(b, a);
    
    
    return (max(num1, num2));
}
