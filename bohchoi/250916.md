**코드(선택)**:

```cpp

#include <string>
#include <vector>

using namespace std;

vector<int> solution(vector<int> arr) {
    vector<int> answer;
    int i = 0;
    
    for (int cnt = 0; cnt < arr.size(); cnt++)
    {
        for (int j = 0; j < arr.at(i); j++)
            answer.push_back(arr[i]);
        i++;
    }
    return answer;
}
```
