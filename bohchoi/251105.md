#include <string>
#include <vector>

using namespace std;

vector<int> solution(string my_string) {
    vector<int> answer(52, 0);
    
    for (int i = 0; i < my_string.size(); i++)
    {
        int tmp = 0;
        if (my_string.at(i) >= 'A' && my_string.at(i) <= 'Z')
        {
            tmp = my_string[i] - 65;
            answer[tmp] += 1;
        }
        else if (my_string.at(i) >= 'a' && my_string.at(i) <= 'z')
        {
            tmp = my_string[i] - 71;
            answer[tmp] += 1;
        }
    }
    return answer;
}
