**오늘의 메모(선택)**:

코드 피드백
1) malloc 크기 계산이 잘못됨
- `malloc` 크기 계산 실수 (`sizeof(size_t)` → `sizeof(int)`)

2) 입력 배열 score를 직접 바꾸고 있음
- 입력 배열 `score` 직접 정렬
- 원본 보존을 위한 `temp` 배열 복사 필요

3) 시간복잡도가 너무 큼
- 매 반복마다 정렬 수행 → 비효율적
- 선택 정렬 직접 구현보다 `qsort()` 활용이 더 적절

- `memcpy(temp, score, (i + 1) * sizeof(int))` : score의 앞부분만 temp로 복사
- `qsort(temp, i + 1, sizeof(int), compare_desc)` : temp의 앞부분을 내림차순 정렬


**코드(선택)**:

```c

//피드백 풀이
#include <stdio.h>
#include <stdbool.h>
#include <stdlib.h>
#include <string.h>

int compare_desc(const void *a, const void *b)
{
    int x = *(const int *)a;
    int y = *(const int *)b;

    if (x > y)
        return -1;
    if (x < y)
        return 1;
    return 0;
}

int* solution(int k, int score[], size_t score_len) {
    int *answer = malloc(score_len * sizeof(*answer));
    int *temp = malloc(score_len * sizeof(*temp));

    if (!answer || !temp)
    {
        free(answer);
        free(temp);
        return NULL;
    }
    
    for (int i = 0; i < score_len; i++)
    {
        memcpy(temp, score, (i+1) * sizeof(*score));
        qsort(temp, i + 1, sizeof(int), compare_desc);
        
        if (i + 1 < k)
            answer[i] = temp[i];
        else
            answer[i] = temp[k - 1];
    }

    free(temp);
    return answer;
}

//내 풀이
#include <stdio.h>
#include <stdbool.h>
#include <stdlib.h>

void select_sort(int sort_array[], int sort_len);
void swap(int sort_array[], int x, int y);

int* solution(int k, int score[], size_t score_len) {
    int* answer = (int*)malloc(score_len * sizeof(size_t));
    
    for (int i = 1; i <= score_len; i++)
    {
        select_sort(score, i);
        
        if (i < k)
            answer[i-1] = score[i - 1];
        else
            answer[i-1] = score[k - 1];
    }
    return answer;
}

void select_sort(int sort_array[], int sort_len)
{
    int max;
    int index;
    for (int i = 0; i < sort_len; i++)
    {
        max = sort_array[i];
        index = i;
        for (int j = i + 1; j < sort_len; j++)
        {
            if (max < sort_array[j])
            {
                max = sort_array[j];
                index = j;
            }
        }       
        swap(sort_array, i, index);
    }
}

void swap(int sort_array[], int x, int y)
{
    int temp = 0;
    temp = sort_array[x];
    sort_array[x] = sort_array[y];
    sort_array[y] = temp;
}
```
