//피드백 풀이
class Solution {
    
    public int solution(String s) {
        String[] english_array = {"zero", "one", "two", "three", "four", "five", "six", "seven", "eight", "nine"};
        
        for (int i = 0 ; i < english_array.length; i++)
        {
            s = s.replaceAll(english_array[i], Integer.toString(i));
        }
    
        return Integer.parseInt(s);;
    }
}

//내 풀이
class Solution {

    public String convert_english_to_number(String s)
    {
        if (s.contains("zero"))
            s = s.replace("zero", "0");
        if (s.contains("one"))
            s = s.replace("one", "1");
        if (s.contains("two"))
            s = s.replace("two", "2");
        if (s.contains("three"))
            s = s.replace("three", "3");
        if (s.contains("four"))
            s = s.replace("four", "4");
        if (s.contains("five"))
            s = s.replace("five", "5");
        if (s.contains("six"))
            s = s.replace("six", "6");
        if (s.contains("seven"))
            s = s.replace("seven", "7");
        if (s.contains("eight"))
            s = s.replace("eight", "8");
        if (s.contains("nine"))
            s = s.replace("nine", "9");
        return s;
    }
    
    public int solution(String s) {
        int answer = 0;
        String answer_string;
        answer_string = convert_english_to_number(s);
        answer = Integer.parseInt(answer_string);

            
        return answer;
    }
}
