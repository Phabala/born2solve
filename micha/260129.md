class Solution {
    public String solution(String code) {
        StringBuilder answer = new StringBuilder();
        int mode = 0;
        char c;
        for (int i = 0; i < code.length(); i++)        
        {
            c = code.charAt(i);
            
            if (mode == '1')
            {
                mode = mode == 0 ? 1 : 0;
                continue;
            }
            
            if (i % 2 == mode)
                answer.append(c);
        }
        
        return answer.length() == 0 ? "EMPTY" : answer.toString();
    }
}
