**오늘의 메모(선택)**:

코드 피드백
1) n = div;
틀린 부분 : 몇 번 교환 했는지임

교환하고 남은 빈 병 n % a
새로 받은 콜라 div * b를 마셔야 생긴 빈 병을 더해야 함

2) b가 answer에만 반영되고, n에 반영 안 됨
div * b 병 받았으면 div *b 빈 병이 생김


**코드(선택)**:

```c

//정답 풀이
#include <stdio.h>
#include <stdbool.h>
#include <stdlib.h>

int solution(int a, int b, int n) {
    int answer = 0;
    int div = 0;
    int mod = 0;
    
    while (n >= a)
    {
        div = n / a;
        mod = n % a;

        answer += b * div;
        n = b * div + mod;
    }
    
    return answer;
}

//오답 풀이
#include <stdio.h> 
#include <stdbool.h> 
#include <stdlib.h> 

int solution(int a, int b, int n) 
{ 
    int answer = 0;
    int div = 0;
    int i = 0;
    int j = 0;
    int mod[1000000];
        
    while (n - a >= 0) {
        div = n / a; 
        if (n % a != 0)
            mod[i++] = n % a;
        n = div;
        if (n - a < 0 && mod[j] && i > j)
            n = n + mod[j++];
        answer += b * div;
    }

    return answer;
}


//오답 풀이 수정
#include <stdio.h> 
#include <stdbool.h> 
#include <stdlib.h> 

int solution(int a, int b, int n) 
{ 
    int answer = 0;
    int div = 0;
        
    while (n - a >= 0) {
        div = n / a; 
        n = (div * b) + (n % a);
        answer += b * div;
    }

    return answer;
}
```
