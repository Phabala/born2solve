class Solution {
    public String solution(String s, int n) {
        String answer = "";
        
        for (int i = 0; i < s.length(); i++)
        {
            char c = s.charAt(i);
            
            if (c >= 'a' && c <= 'z')
            {
                c = (char)((c - 'a' + n) % 26 + 'a');
            }
            else if (c >= 'A' && c <= 'Z')
            {
                c = (char)((c - 'A' + n) % 26 + 'A');
            }
            
            answer += c;
        }
        
        return answer;
    }
}

class Solution {
    public String solution(String s, int n) {
        String answer = "";
        
        for (int i = 0; i < s.length(); i++)
        {
            char c = s.charAt(i);
            
            if (c == 'z' || c == 'Z')
            {
                if (n == 1)
                    c = (char)(c - 25);
                else
                    c = (char)(c % 25 + n - 1);
            }
            else if (c >= 'a' && c <= 'z')
            {
                if (n + (c - 'a')  > 25)
                {
                    n = n % 25;
                }
                c = (char)(c + n);
            }
            else if (c >= 'A' && c <= 'Z')
            {
                if (n + (c - 'A')  > 25)
                {
                    n = n % 25;
                }
                c = (char)(c + n);
            }
            
            answer += c;
        }
        
        return answer;
    }
}
