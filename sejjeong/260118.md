```C++

#include <string>
#include <vector>
#include <unordered_map>

using namespace std;

int solution(vector<int> topping) {
    int answer = 0;
    
    // 각 조각에 동일한 가짓 수의 토핑이 올라가면 공평하게 나눠진 것
    // 어딜 반으로 나눌 것인가
    // 100만번 짤라야 해
    // 오른쪽 토핑 왼쪽으로 옮기면 되잖아.
    
    unordered_map<int, int> leftToppings;
    unordered_map<int, int> rightToppings;

    for (size_t i = 0; i < topping.size(); ++i)
    {
        ++rightToppings[topping[i]];
    }
    
    for (size_t i = 0; i < topping.size(); ++i)
    {
        ++leftToppings[topping[i]];
        --rightToppings[topping[i]];
        if (rightToppings[topping[i]] == 0)
        {
            rightToppings.erase(topping[i]);
        }
        if (leftToppings.size() == rightToppings.size())
        {
            ++answer;
        }
    }
    
    return answer;
}
```
