```C++
#include <string>
#include <vector>
#include <unordered_map>
#include <map>

using namespace std;

vector<int> solution(string s) {
    // 중복되는 원소가 없으면 {}로 표현할 수 있다고 함
    // 빈도수에 맞게 정렬해야 하네
    // 아 내가 파싱해야 하네;; 
    // 길이는 10만 이하니까
    
    // 반복문 한 번 돌면서, 
    // 중괄호 열기, 닫기, 반점이 보이면 건너뛰기 

    unordered_map<int, int> m;
    
    size_t i = 0;
    while (i < s.size())
    {
        if (s[i] == '{' || s[i] == '}' || s[i] == ',')
        {
            ++i;
            continue;
        }
        
        size_t j = i;
        size_t value = 0;
        while (j < s.size() && s[i] != '{' && s[i] != '}' && s[i] != ',')
        {
            value *= 10;
            value += s[j] - '0';
            ++j;
            ++i;
        }
        ++m[value];
    }
    
    
    std::map<int, int> reverse;
    for (auto pair : m)
    {
        reverse[pair.second] = pair.first;
    }
    
    vector<int> temp;
    for (auto pair : reverse)
    {
        temp.push_back(pair.second);    
    }
    
    vector<int> answer;
    for (int i = temp.size() - 1; i >= 0; --i)
    {
        answer.push_back(temp[i]);
    }

    
    return answer;
}
```
