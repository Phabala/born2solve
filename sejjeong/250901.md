```
#include <vector>
#include <unordered_set>
using namespace std;

int solution(vector<int> nums)
{
    unordered_set<int> set;
    
    // 중복 제거
    for (size_t i = 0; i < nums.size(); ++i)
    {
        set.insert(nums[i]);
    }
    
    int n = set.size();
    int r = nums.size() / 2;

    int answer = n > r ? r : n;
    return answer;
}

```
