#include <iostream>
#include <string>
#include <cstdint>

using namespace std;

bool IsPalindrome(const char* start, const char* end)
{
    while (start < end)
    {
        if (*start != *end)
        {
            return false;
        }
        ++start;
        --end;
    }
    return true;
}

int solution(string s)
{
    int answer = 1;  // 최소 1임
    
    // 2500 ^2 괜찮네
    int32_t i = 0;
  	const char* str = s.c_str();
    while (i < s.size())
    {
        int32_t j = s.size() - 1;
        while (i < j)
        {
			if (IsPalindrome(str + i, str + j))
        	{
                answer = std::max(answer, j - i + 1);
            	break;
        	}
            --j;
        }
        ++i;
    }

    return answer;
}
