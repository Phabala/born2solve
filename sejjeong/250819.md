**오늘의 메모(선택)**: 통과는 하였지만 조잡하고 속도도 많이 느림, 사람들이 제출한 답안을 보니 모두 투포인터를 사용
                    
**코드(선택)**:

C++

```
#include <algorithm>
#include <vector>
#include <map>
#include <iostream>
class Solution {
public:
    vector<vector<int>> threeSum(vector<int>& nums) {
        
        // 3개의 합이 0인 조합 집합 만들기
        set<vector<int>> set;

        map<int, int> map; 
     
        for (size_t i = 0; i < nums.size(); ++i)
        {
            if (map[nums[i]] != 0)
            {
                ++map[nums[i]];
            }
            else
            {
                map[nums[i]] = 1;
            }
        }

        for (auto pair : map)
        {
            for (auto pair2 : map)
            {
                if (pair.first == pair2.first && pair.second == 1)
                {
                    continue;
                }
               
                int sum = pair.first + pair2.first;
                if (map.find(0 - sum) != map.end())
                {
                    if ((0 - sum == pair.first && pair.second == 1)
                    || (0 - sum == pair2.first && pair2.second == 1)
                    || 0 - sum == pair2.first && 0 - sum == pair.first && pair.second <= 2)
                    {
                        continue;
                    }

                    vector<int> temp;
                    if (sum < 0)
                    {
                        temp.push_back(pair.first);
                        temp.push_back(pair2.first);
                        temp.push_back(0 - sum);
                    }
                    else
                    {
                        temp.push_back(0 - sum);
                        temp.push_back(pair.first);
                        temp.push_back(pair2.first);
                    }
                    std::sort(temp.begin(), temp.end());
                    set.insert(temp);
                }

            }
        }

        vector<vector<int>> result;
        for (auto ele : set)
        {
            result.push_back(ele);
        }
        return result;
    }
};
```
