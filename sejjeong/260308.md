```C++
#include <string>
#include <vector>
#include <cstdint>
#include <unordered_map>
#include <algorithm>

using namespace std;

bool findPath(vector<string>& answer, const uint32_t remainPort, unordered_map<string, vector<string>>& adjList, const string& start)
{
    vector<string>& to = adjList[start];
    for (uint32_t i = 0; i < to.size(); ++i)
    {
        string port = to[i];
        
        answer.push_back(port);
        to.erase(to.begin() + i);
        
        if (findPath(answer, remainPort - 1, adjList, port))
        {
            return true;
        }
        
        answer.pop_back();
        to.insert(to.begin() + i, port);
    }
    return remainPort == 0;
}

vector<string> solution(vector<vector<string>> tickets) {
    
    // assert 안 됨
    // tickets[0][0] == "ICN"; 일거라 예상
    
    // [0] to [1]
    // 방향 그래프, 10000개니까 인접 배열보단 인접 리스트가 더 적합함
    // ICN에서 출발
    // 다 돌아야 한다고? DFS가 맞겠네
    
    unordered_map<string, vector<string>> adjList;
    for (uint32_t i = 0; i < tickets.size(); ++i)
    {
        vector<string>& ticket = tickets[i];
        adjList[ticket[0]].push_back(ticket[1]);
    }
    
    for (auto& pair : adjList)
    {
        vector<string>& adj = pair.second;
        sort(adj.begin(), adj.end());
    }
    
    // 무엇을 넣고 무엇을 빼야하나?
    // INC로 시작하기
    // 반복문 돌면서 목적지 가는 것 호출하기
    // 만약에 실패하면 돌아올 때, 값 원래대로 되돌리기 
    // stack 2개 쓰거나, 재귀가 나음
    // 깊어도 1만인데, 플로이드 워셜이랑 복잡도 똑같지 않나? 
    
    // count도 전달, 항공권 다 사용했는지 체크
    
    vector<string> answer; // stack으로 해?
    answer.push_back("ICN");
    findPath(answer, tickets.size(), adjList, "ICN");
    return answer;
}
```
