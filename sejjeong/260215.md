**코드(선택)**:



```C++

#include <string>
#include <vector>
#include <algorithm>
#include <iostream>

using namespace std;

// 0215, 1835 시작
// 0215, 1940 휴식
//       2145~2155 해결

bool compare(const vector<int>& a, const vector<int>& b)
{
    return a[0] < b[0];
}

int solution(vector<vector<int>> routes) {
    
    // 10000대 이하, n log n 까지 sort일 듯
    // 경로는 선형적임 -30,000 ~ 30,000
    
    // 모든 차량은 단속용 카메라를 한 번은 만나야 한다.
    // -> sort는 진입 시점일 것, 제일 빨리 들어온 애가 나가기 전에 카메라 설치하거나, 
    // 중간에 먼저 나가야 하는 애가 있으면 그 때 설치해야 한다.
    // 이후 진출 시점보다 늦은 진입 시점을 가진애부터 다시 판단하기
    
    std::sort(routes.begin(), routes.end(), compare);
    // for (int i = 0; i < routes.size(); ++i)
    // {
    //     std::cout << routes[i][0] << ", " << routes[i][1] << std::endl;
    // }
    
    int answer = 0;
    size_t i = 0; 
    while (i < routes.size())
    {
        ++answer;
        size_t j = i + 1;
        // 아 최저 exit를 정하고 최저 exit 보다 느리면 다음으로 넘어가기
        int minExit = routes[i][1];
        while (j < routes.size()) 
        {
            int currentExit = routes[j][1];
            minExit = min(currentExit, minExit);
            int currentEnter = routes[j][0];
            if (minExit < currentEnter)
            {
                break;
            }
            ++j;
        }
        
        i = j;
    }
    return answer;
}
```
