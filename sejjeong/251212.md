**오늘의 메모(선택)**: 
            진짜 평범한 배낭 문제

**코드(선택)**:



```C++


#include <iostream>
#include <vector>

using namespace std;

int main(void)
{
    int n;
    int capacity;
    std::cin >> n >> capacity;
    
    vector<int> weights;
    vector<int> values;
    vector<vector<int>> grid;
    for (int i = 0; i < n; ++i)
    {
        int weight;
        int value;
        std::cin >> weight >> value;
        weights.push_back(weight);
        values.push_back(value);
        grid.emplace_back(vector<int>(capacity, 0));
    }
    
    for (int i = weights[0] - 1; i < capacity; ++i)
    {
        grid[0][i] = values[0];
    }
    
    for (int i = 1; i < n; ++i)
    {
        int currentWeight = weights[i];
        for (int j = 0; j < currentWeight - 1 && j < capacity; ++j)
        {
            grid[i][j] = grid[i - 1][j];
        }
        int currentValue = values[i];
        for (int j = currentWeight - 1; j < capacity; ++j)
        {
            int beforeMaxValue = grid[i - 1][j];
            int remainingWeight = j - (currentWeight - 1);
            int beforeRemainingWeightMaxValue = remainingWeight != 0 ? grid[i - 1][remainingWeight - 1] : 0;
            if (beforeMaxValue < currentValue + beforeRemainingWeightMaxValue)
            {
                grid[i][j] = currentValue + beforeRemainingWeightMaxValue;
            }
            else
            {
                grid[i][j] = beforeMaxValue;
            }
        }
    }
    
    std::cout << grid[n - 1][capacity - 1];
    
    return 0;
}
```
