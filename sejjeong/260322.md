                뒤에 있는 큰 수 찾기(Lv2, 64%)

**오늘의 메모(선택)**: 
            힌트 보고 겨우 풀었다. stack은 큰 것만 남는 것을 이용


**코드(선택)**:



```C++
#include <string>
#include <vector>

using namespace std;

vector<int> solution(vector<int> numbers) {
    // 자신보다 뒤에 있는 숫자 중, 나보다 크면서 가장 가까이 있는 수를
    // 뒷 큰수
    
    // 100만임
    // n log n 안에 끝낼 것
    vector<int> answer(numbers.size(), -1);
    vector<int> temp; // back이 제일 큼
    for (int32_t i = 0; i < numbers.size(); ++i)
    {
        while (temp.size() != 0)
        {
            if (numbers[temp.back()] >= numbers[i])
            {
                break;
            }
            else
            {
                answer[temp.back()] = numbers[i];
                temp.pop_back();
            }
        }
        temp.push_back(i);
    }
    
    return answer;
}
```
