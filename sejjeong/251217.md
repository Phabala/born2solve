```C++

#include <string>
#include <vector>
#include <list>
#include <algorithm>

using namespace std;

string toLower(const string& str)
{
    string result;
    for (int i = 0; i < str.size(); ++i)
    {
        result += str[i] | 0x20;
    }
    return result;
}

int solution(int cacheSize, vector<string> cities) {
    int answer = 0;
    // 캐시 사이즈가 작으니 linkedList로 해도 됨
    
    if (cacheSize == 0)
    {
        return cities.size() * 5;
    }
    
    list<string> cache;
    
    for (int i = 0; i < cities.size(); ++i)
    {
        string lowercase = toLower(cities[i]);
        auto it = find(cache.begin(), cache.end(), lowercase);
        if (it == cache.end())
        {
            if (cache.size() == cacheSize)
            {
                cache.erase(--cache.end());
            }
            answer += 5;
        }
        else
        {
            cache.erase(it);
            ++answer;
        }
        cache.push_front(lowercase);
    }
    
    return answer;
}
```
