**오늘의 메모(선택)**: 

**코드(선택)**:


```
#include <climits>

class Solution {
public:
    int reverse(int x) {
        
        long result = 0;

        while (x != 0)
        {
            result *= 10;
            result += x % 10;
            x /= 10;
        }
        return result > INT_MAX || result < INT_MIN ? 0 : result;
    }
};

```
