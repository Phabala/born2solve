**오늘의 메모(선택)**: 메모리 사용량이 많이 나옴, 알고보니 다른 통과자들은 동적 할당 및 복사를 하지 않고 인자들끼리 병합함
                    두 가지 버전을 작성
                    1. 주먹구구식
                    2. 더미 노드를 이용하여 조건 최소화 및 루프 안에 조건문을 밖으로 빼내어 최적화

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */

// 최적화 버전
class Solution {
public:
    ListNode* mergeTwoLists(ListNode* list1, ListNode* list2) {

        ListNode head;
        ListNode* tail = &head;
 
        while (list1 != nullptr && list2 != nullptr)
        {
            if (list2->val <= list1->val)
            {
                pushNode(&tail, list2->val);
                list2 = list2->next;                
            }
            else
            {
                pushNode(&tail, list1->val);
                list1 = list1->next;
            }
        }

        if (list1 == nullptr)
        {
            while (list2 != nullptr)
            {
                pushNode(&tail, list2->val);
                list2 = list2->next;
            }
        }
        else if (list2 == nullptr)
        {
            while (list1 != nullptr)
            {
                pushNode(&tail, list1->val);
                list1 = list1->next;
            }
        }
        return head.next;
    }

    void pushNode(ListNode** tail, int val)
    {
        if (*tail == nullptr)
        {
            *tail = new ListNode(val);
            return;
        }
        ListNode* back = new ListNode(val);
        (*tail)->next = back;
        *tail = back;
    }
};


 // 주먹구구식
class Solution {
public:
    ListNode* mergeTwoLists(ListNode* list1, ListNode* list2) {

        if (list1 == nullptr && list2 == nullptr)
        {
            return nullptr;
        }

        ListNode* head = nullptr;
        ListNode* tail = nullptr;
        if (list1 == nullptr)
        {
            pushNode(&tail, list2->val);
            head = tail;
            list2 = list2->next;
        }
        else if (list2 == nullptr)
        {
            pushNode(&tail, list1->val);
            head = tail;
            list1 = list1->next;
        }
        else
        {
            if (list2->val <= list1->val)
            {
                pushNode(&tail, list2->val);
                head = tail;
                list2 = list2->next;
            }
            else
            {
                pushNode(&tail, list1->val);
                head = tail;
                list1 = list1->next;
            }
        }
 

        while (list1 != nullptr || list2 != nullptr)
        {
            if (list1 == nullptr)
            {
                pushNode(&tail, list2->val);
                list2 = list2->next;
            }
            else if (list2 == nullptr)
            {
                pushNode(&tail, list1->val);
                list1 = list1->next;
            }
            else
            {
                if (list2->val <= list1->val)
                {
                    pushNode(&tail, list2->val);
                    list2 = list2->next;                
                }
                else
                {
                    pushNode(&tail, list1->val);
                    list1 = list1->next;
                }
            }
        }
        return head;
    }

    void pushNode(ListNode** tail, int val)
    {
        if (*tail == nullptr)
        {
            *tail = new ListNode(val);
            return;
        }
        ListNode* back = new ListNode(val);
        (*tail)->next = back;
        *tail = back;
    }
};
