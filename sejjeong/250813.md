#include <cstring>
class Solution {
public:
    int romanToInt(string s) {
        
        int romanInt = 0;
        const char* str = s.c_str();
        int i = 0;
        while (*str != '\0')
        {
            if (std::strncmp("CM", str, 2) == 0)
            {
                romanInt += 900;
                str += 2;
            }
            else if (std::strncmp("CD", str, 2) == 0)
            {
                romanInt += 400;
                str += 2;
            }
            else if (std::strncmp("XC", str, 2) == 0)
            {
                romanInt += 90;
                str += 2;
            }
            else if (std::strncmp("XL", str, 2) == 0)
            {
                romanInt += 40;
                str += 2;
            }
            else if (std::strncmp("IX", str, 2) == 0)
            {
                romanInt += 9;
                str += 2;
            }
            else if (std::strncmp("IV", str, 2) == 0)
            {
                romanInt += 4;
                str += 2;
            }
            else
            {
                if (*str == 'I')
                {
                    romanInt += 1;
                }
                else if (*str == 'V')
                {
                    romanInt += 5;
                }
                else if (*str == 'X')
                {
                    romanInt += 10;
                }
                else if (*str == 'L')
                {
                    romanInt += 50;
                }
                else if (*str == 'C')
                {
                    romanInt += 100;
                }
                else if (*str == 'D')
                {
                    romanInt += 500;
                }
                else if (*str == 'M')
                {
                    romanInt += 1000;
                }
                ++str;
            }
        }
        return romanInt;
    }
};
