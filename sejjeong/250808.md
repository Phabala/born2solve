```
#include <climits>
class Solution {
public:
    int myAtoi(string s) {
        
        long result = 0;
        const char* cstr = s.c_str();

        while (*cstr != '\0' && *cstr == ' ')
        {
            ++cstr;
        }

        if (*cstr == '-')
        {
            isNegative = true;
            ++cstr;
        }
        else if (*cstr == '+')
        {
            ++cstr;
        }

        while (*cstr != '\0' && *cstr >= '0' && *cstr <= '9' && result <= 2147483648L)
        {
            result = result * 10 + *cstr - '0';
            ++cstr;
        }

        result *= isNegative ? -1 : 1;
        if (result > INT_MAX)
        {
            result = INT_MAX;
        }
        if (result < INT_MIN)
        {
            result = INT_MIN;
        }
        return result;
    }
};

```
