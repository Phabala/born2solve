#include <cassert>
#include <cstring>
#include <deque>
#include <string>

using namespace std;


class Solution {
    #define MAX(num1, num2) (num1 > num2 ? num1 : num2)
public:
    string simplifyPath(string path) {

        assert(path.size() >= 1);

        deque<string> simplifyPath;

        const char* cPath = path.c_str() + 1;

        while (*cPath != '\0')
        {
            while (*cPath == '/')
            {
                ++cPath;
            }

            const char* end = strchr(cPath, '/');
            size_t size;
            if (end != nullptr)
            {
                size = end - cPath;
            }
            else
            {
                size = strlen(cPath);
            }
            
            if (strncmp(cPath, "..", MAX(size, 2)) == 0)
            {
                if (simplifyPath.size() >= 1)
                {
                    simplifyPath.pop_back();
                }
            }
            else if (strncmp(cPath, ".", size) == 0)
            {

            }
            else
            {
                string name = string(cPath, size);
                simplifyPath.push_back(name);
            }

            cPath += size;

            while (*cPath == '/')
            {
                ++cPath;
            }
        }

        if (simplifyPath.empty())
        {
            return "/";
        }

        string result;
        while (simplifyPath.size() != 0)
        {
            result += "/";
            result += simplifyPath.front();
            simplifyPath.pop_front();
        }
        return result;
    }
};
