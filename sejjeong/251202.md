#include <string>
#include <vector>
#include <iostream>
#include <unordered_set>
#include <unordered_map>

using namespace std;

int findCombinationRecursive(const int k, const int depth, const vector<int>& clothes, int count, const int startIndex, int total)
{
    if (k == depth)
    {
        return 0;
    }
    for (int i = startIndex; i < clothes.size(); ++i)
    {
        count *= clothes[i];
        total += count;
        total += findCombinationRecursive(k, depth + 1, clothes, count, i + 1, 0);
        count /= clothes[i];
    }
    return total;
}


int findCombination(const int k, const vector<int>& clothes)
{
    return findCombinationRecursive(k, 0, clothes, 1, 0, 0);
}

int solution(vector<vector<string>> clothes) {
    
    unordered_map<string, int> map;
    for (int i = 0; i < clothes.size(); ++i)
    {
        ++map[clothes[i][1]];
    }
    
    vector<int> clotheCounts;
    for (auto pair : map)
    {
        clotheCounts.push_back(pair.second);
    }
    
    int result = findCombination(clotheCounts.size(), clotheCounts);
 
    
    
    // 최소 한 개는 입음
    
    // [이름, 종류]
    // 종류 개수
    // 조합 고르기 1개 조합 개수, 2개 조합 개수
    // index랑 개수로 매칭해놨어
    // 각각 k개랑 
    
    return result;
}
