import  sys
from itertools import permutations


def getPermOrderCnt(a, b, c):
    valList = []
    for _ in range(a) :
        valList.append(3)
    for _ in range(b) :
        valList.append(2)
    for _ in range(c) :
        valList.append(1)
    
    tmpList = []
    orderCnt = 0
    for perm in permutations(valList) :
        if perm in tmpList :
            continue
        tmpList.append(perm)
        orderCnt += 1
    return (orderCnt)

def getOrderCnt(n) :
    cnt = 0
    for a in range(n//3 + 1) :
        for b in range((n - 3 * a)// 2 + 1) :
            for c in range(n - (3 * a) - (2 * b)+ 1) :
                if (n == (3 * a + 2 * b + c)) :
                    cnt += getPermOrderCnt(a, b, c)
    return (cnt)

t = int(sys.stdin.readline().strip())
inputVal = [ int(sys.stdin.readline().strip()) for _ in range(t)]

for i in range(t):
    print(getOrderCnt(inputVal[i]))
