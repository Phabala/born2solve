def solution(S):
    if not S:
        return 1

    myDict = {"}" : "{", ")" : "(", "]" : "["}

    stack = []
    for c in S:
        if c == "{" or c == "(" or c == "[":
            stack.append(c)
        else:
            if not stack or stack[-1] != myDict[c]:
                return 0
            stack.pop()
    
    if stack:
        return 0
    else:
        return 1
\```
