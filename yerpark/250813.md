**오늘의 메모(선택)**: 

**코드(선택)**:

```c++
#include <iostream>

int    main(void)
{
    int    n;
    
    std::cin >> n;
    
    for (int i = 1; i <= n; i++)
    {
        for (int j = 1; j <= i; j++)
        {
            std::cout <<"*";
        }
        std::cout <<"\n";
    }
    for (int i = n - 1; i >= 1; i--)
    {
        for (int j = 1; j <= i; j++)
        {
            std::cout <<"*";
        }
        std::cout <<"\n";
    }
    return (0);
}
\```
