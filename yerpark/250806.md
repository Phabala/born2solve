**문제 이름 또는 번호**: 28278 / 스택 문제

**문제 링크**: https://www.acmicpc.net/problem/28278

**오늘의 메모(선택)**: endl과 \n의 차이에 대해 느꼈습니다.. 

**코드(선택)**:

```c++
#include <stack>
#include <iostream>

int    main(void)
{
    std::ios::sync_with_stdio(false);
    std::cin.tie(NULL);
    
    int                 n;
    int                 tmp1, tmp2;
    std::stack<int>     s;
    
    std::cin >> n;
    
    for (int i = 0; i < n; i++)
    {
        std::cin >> tmp1;
        if (tmp1 == 1)
        {
            std::cin >> tmp2;
            s.push(tmp2);
        }
        else if (tmp1 == 2)
        {
            if (s.empty() == true)
                std::cout << "-1\n";
            else
            {
                std::cout << s.top() << "\n";
                s.pop();
            }
        }
        else if (tmp1 == 3)
        {
            std::cout << s.size() << "\n";
        }
        else if (tmp1 == 4)
        {
            if (s.empty() == true)
                std::cout << 1 << "\n";
            else
                std::cout << 0 << "\n";
        }
        else if (tmp1 == 5)
        {
            if (s.empty() == true)
                std::cout << -1 << "\n";
            else
            {
                std::cout << s.top() << "\n";
            }
        }
        else
        {
            std::cerr << "error\n";
        }
    }
    
    return (0);
}
\```
