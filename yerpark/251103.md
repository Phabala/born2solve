import sys

# 수열의 값과 스택의 탑을 비교 
    # 만약 수열의 값이 스택의 탑보다 크다면
        # 실패
    # 같다면, 해당 값 pop
    # 작다면
        # 같아질때까지 push
def doAllPush(myStack, printList, number):
    myStack.append(number)
    printList.append('+')
    return 1

def doAllPop(myStack, printList):
    myStack.pop()
    printList.append('-')

if __name__ == "__main__":
    n = int(sys.stdin.readline().strip())
    myStack = []
    printList = []
    number = 1
    okFlag = True

    for _ in range(n):
        cur = int(sys.stdin.readline().strip())
        while (okFlag):
            if len(myStack) == 0 or myStack[-1] < cur:
                number += doAllPush(myStack, printList, number)
            elif myStack[-1] == cur:
                doAllPop(myStack, printList)
                break
            elif myStack[-1] > cur:
                print("NO")
                okFlag = False
        if (okFlag == False):
            break
    if (okFlag):
        for a in printList:
            print(a)
