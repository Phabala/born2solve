**오늘의 메모(선택)**: 

**코드(선택)**:

```c++
#include <iostream>

int    main(void)
{
    int    n;
    
    std::cin >> n;
    
    for (int i = 1; i <= n; i++)
    {
        std::string    str;
        if (i % 2 != 0)
            str = "* ";            
        else
            str = " *";
        for (int j = 0; j < n; j++)
            std::cout << str;
        std::cout << "\n";
    }
    return (0);
}
\```
