def solution(n, bans):
    alphabet = "abcdefghijklmnopqrstuvwxyz"
    alphabet_dict = {alphabet[i]: i for i in range(26)}
    bans.sort()

    def digits(x):
        d = 0
        while x > 0:
            d += 1
            x = (x - 1) // 26 #z인 경우 고려 
        return d
    
    # 문자열을 엑셀식 숫자로 변환 (a=1..a=26)
    def s2n(s):
        v = 0
        for ch in s:
            v = v * 26 + (alphabet_dict[ch] + 1)
        return v 

    #숫자를 문자열로 변환
    def n2s(x):
        res = ""
        while x > 0 :
            res = alphabet[(x-1) % 26] + res
            x = (x - 1) // 26
        return res
    
    target = n
    while True:
        str_len = digits(target)
        deleted_cnt = 0

        # 기준은 target 이하인 ban 개수
        for s in bans:
            if len(s) < str_len:
                deleted_cnt += 1
            elif len(s) == str_len:
                if s2n(s) <= target:
                    deleted_cnt += 1
            else:
                continue
    
        new_target = n + deleted_cnt
        if new_target == target:
            break
        target = new_target

    return n2s(target)


\```
