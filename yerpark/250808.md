**코드(선택)**:

```c++
#include <iostream>
#include <stack>
#include <string>

int    main(void)
{
    int                 t;
    
    std::cin >> t;
    for (int i = 0; i < t; i++)
    {
        std::stack <char>    s;
        std::string          str;
        int                  errFlag = 0;
        
        std::cin >> str;
        for (int j = 0; j < str.length(); j++)
        {
            if (str[j] == '(')
                s.push('(');
            else
            {
                if (s.empty() == true)
                {
                    errFlag = 1;
                    break ;
                }
                else
                    s.pop();
            }
        }
        if (s.empty() == true && errFlag == 0)
            std::cout << "YES\n";
        else
            std::cout << "NO\n";
    }

    return (0);
}
