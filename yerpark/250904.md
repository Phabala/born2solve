**코드(선택)**:

```c++
#include <iostream>

bool    isVowel(char a)
{
    if (a == 'A' || a == 'E' || a == 'I' || a == 'O' || a == 'U')
        return (true);
    if (a == 'a' || a == 'e' || a == 'i' || a == 'o' || a == 'u')
        return (true);
    return (false);
}


int    main(void)
{
    int    n;
    
    std::cin >> n;
    std::cin.ignore();
    
    for (int i = 0; i < n; i++)
    {
        int c_cnt = 0, v_cnt = 0;
        std::string    str;
        
        std::getline(std::cin, str);
        
        for (int j = 0; j < str.length(); j++)
        {
            if (isVowel(str[j]) == true)
            {
                v_cnt++;
            }
            else if (str[j] != ' ')
            {
                c_cnt++;
            }
        }
        std::cout << c_cnt << " " << v_cnt << std::endl ;
    }
    return (0);
}
\```
