```python3
import sys
import heapq

if __name__ == "__main__":
    n = int(sys.stdin.readline().strip())
    myHeap = []

    for _ in range(n):
        classNumber, startTime, endTime = map(int, sys.stdin.readline().split())
        heapq.heappush(myHeap, [startTime, endTime, classNumber])

    room = [] #각 원소 -> room : [lastEndTime, lastStartTime]
    heapq.heappush(room, [0, 0, 1])
    roomCount = 1
    class_room = [0] * (n + 1)

    while (myHeap):
        currClass = heapq.heappop(myHeap)

        if (room[0][0] <= currClass[0]):
            currRoom = heapq.heappop(room)
            currRoom[0] = currClass[1]
            currRoom[1] = currClass[0]
            class_room[currClass[2]] = currRoom[2]
            heapq.heappush(room, currRoom)
        else:
            roomCount += 1
            newRoom = [currClass[1], currClass[0], roomCount]
            class_room[currClass[2]] = roomCount
            heapq.heappush(room, newRoom)

    print(len(room))

    for i in range(1, n + 1):
        print(class_room[i])
