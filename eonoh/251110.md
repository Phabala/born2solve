#include <iostream>
#include <algorithm>
#include <limits.h>
using namespace std;

int n, minn;
int *s;

void init() {
	s = new int[n * n]();
	minn = INT_MAX;
}

int main() {
	ios::sync_with_stdio(false); cin.tie(NULL);

	int t, tmp, value;
	cin >> t;

	for (int tc = 1; tc <= t; tc++) {
		cin >> n;
		init();
		for (int y = 0; y < n; y++) {
			for (int x = 0; x < n; x++) {
				cin >> tmp;
				if (y < x) s[y * n + x] = tmp;
				else s[x * n + y] += tmp;
			}
		}

	}
	return 0;
}
