#include <iostream>
#include <cstring>
#include <limits.h>
#include <queue>
#include <string>
using namespace std;

int t, n, result;
int map[100][100];
int dist[100][100];
int dy[4] = { -1, 1, 0, 0 };
int dx[4] = { 0, 0, -1, 1 };

void dijkstra() {
	int py, px, pCost;
	int ny, nx, nCost;
	priority_queue<pair<int, pair<int, int>>> pq;

	pq.push({0, {0, 0}});
	dist[0][0] = 0;
	while (!pq.empty()) {
		auto cur = pq.top();
		pq.pop();
		
		py = cur.second.first, px = cur.second.second, pCost = dist[py][px];
		for (int i = 0; i < 4; i++) {
			ny = py + dy[i], nx = px + dx[i];
			if (ny < 0 || ny >= n || nx < 0 || nx >= n) continue;
			nCost = pCost + map[ny][nx];
			if (nCost < dist[ny][nx]) {
				dist[ny][nx] = nCost;
				pq.push({-nCost, {ny, nx} });
			}
		}
	}
}

int main()
{
	ios::sync_with_stdio(false);
	cin.tie(NULL);

	cin >> t;
	for (int tc = 1; tc <= t; tc++) {
		cin >> n;
		for (int y = 0; y < n; y++) {
			string str;
			cin >> str;
			for (int x = 0; x < n; x++) {
				map[y][x] = str[x] - '0';
				dist[y][x] = INT_MAX;
			}
		}
		dijkstra();
		cout << "#" << tc << " " << dist[n - 1][n - 1] << "\n";
	}
}
