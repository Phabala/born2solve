오늘의 메모(선택): 2의 제곱근이 아니어따 ..

코드(선택): cpp

```cpp
#include <iostream>
#include <cmath>
using namespace std;

long long n, tmp, result;

void init() {
	result = 0;
}

void solve() {
	while (n != 2) {
		tmp = sqrt(n);
		// 제곱근일 경우
		if (tmp * tmp == n) {
			result++;
			n = tmp;
		}
		// 제곱근이 아닐경우 가까운 제곱근까지 더하기
		else {
			tmp = (tmp + 1) * (tmp + 1);
			result += (tmp - n);
			n = tmp;
		}
	}
}

int main() {
	ios::sync_with_stdio(false); cin.tie(NULL);

	int t;
	cin >> t;

	for (int tc = 1; tc <= t; tc++) {
		init();
		cin >> n;
		solve();
		cout << "#" << tc << " " << result << "\n";
	}
	return 0;
}
```
