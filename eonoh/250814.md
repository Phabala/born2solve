#include <iostream>
#include <vector>
using namespace std;

vector<vector<int>> map;
vector<vector<int>> cost;
vector<vector<int>> maxCost;
vector<vector<int>> Dp;

int n, m, c, result;

void init()
{
    map.assign(n, vector<int>(n, 0));
    cost.assign(n, vector<int>(n, 0));
    maxCost.assign(n, vector<int>(n, 0));
    Dp.assign(n, vector<int>(n, 0));
    result = 0;
}

void getMaxCost(int x, int y, int nowX, int sum, int profit)
{
    if (nowX == x + m)
    {
        maxCost[y][x] = max(profit, maxCost[y][x]);
        return;
    }
    int pickedHoneyQuantity = map[y][nowX];
    int pickedHoneyCost = cost[y][nowX];
    if (pickedHoneyQuantity + sum <= c)
    {
        getMaxCost(x, y, nowX + 1, sum + pickedHoneyQuantity, profit + pickedHoneyCost);
    }
    getMaxCost(x, y, nowX + 1, sum, profit);
}

void setDp()
{
    int prev = 0;
    for (int y = n - 1; y >= 0; y--)
    {
        for (int x = n - 1; x >= 0; x--)
        {
            Dp[y][x] = max(prev, maxCost[y][x]);
            prev = Dp[y][x];
        }
    }
}

void solve()
{
    for (int y1 = 0; y1 < n; y1++)
    {
        for (int x1 = 0; x1 < n - m + 1; x1++)
        {
            int y2 = y1;
            int x2 = x1 + m;
            if (x2 >= n)
            {
                y2 = y1 + 1;
                x2 = 0;
            }
            if (y2 == n)
            {
                return;
            }
            int firstWorkerCost = maxCost[y1][x1];
            int secondWorkerCost = Dp[y2][x2];
            result = max(result, firstWorkerCost + secondWorkerCost);
        }
    }
}

int main()
{
    int t;

    cin >> t;
    for (int i = 0; i < t; i++)
    {
        cin >> n >> m >> c;
        init();
        for (int y = 0; y < n; y++)
        {
            for (int x = 0; x < n; x++)
            {
                cin >> map[y][x];
                cost[y][x] = map[y][x] * map[y][x];
            }
        }
        for (int y = 0; y < n; y++)
        {
            for (int x = 0; x < n - m + 1; x++)
            {
                getMaxCost(x, y, x, 0, 0);
            }
        }
        setDp();
        solve();
        cout << '#' << i + 1 << ' ' << result << '\n';
    }
    return 0;
}
