#include <iostream>
#include <algorithm>
#include <limits.h>
using namespace std;
 
int n, b, result, minn;
int height[20];
 
void init() {
	minn = INT_MAX;
}
 
void subset() {
	for (int i = 0; i < (1 << n); i++) {
		result = 0;
		for (int j = 0; j < n; j++) {
			if (i & (1 << j)) result += height[j];
		}
		if (result >= b) minn = min(minn, result);
	}
}
 
int main() {
	ios::sync_with_stdio(false); cin.tie(NULL);
	int t;
 
	cin >> t;
	for (int tc = 1; tc <= t; tc++) {
		init();
		cin >> n >> b;
		for (int i = 0; i < n; i++) {
			cin >> height[i];
		}
		subset();
		cout << "#" << tc << " " << minn - b << "\n";
	}
	return 0;
}
