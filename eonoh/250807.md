문제 이름 또는 번호: 11724 / 연결 요소의 개수

문제 링크: https://www.acmicpc.net/problem/11724

오늘의 메모(선택): dfs 문제입니다.

코드(선택): cpp

```cpp
#include <iostream>
#include <vector>
using namespace std;

vector<vector<int> > graph(1001);
vector<bool> isVisited(1001);

void dfs(int start) {
	isVisited[start] = true;
	for (int next : graph[start]) {
		if (!isVisited[next]) {
			dfs(next);
		}
	}
}

int main() {
	int n, m, u, v, cnt = 0;

	cin >> n >> m;
	for (int i = 0; i < m; i++) {
		cin >> u >> v;
		graph[u].push_back(v);
		graph[v].push_back(u);
	}
	for (int i = 1; i < n + 1; i++) {
		if (!isVisited[i]) {
			dfs(i);
			cnt++;
		}
	}
	cout << cnt << "\n";
	return 0;
}
```
