문제 플랫폼: 백준

문제 이름 또는 번호: 15649 / N과 M(1)

문제 링크: https://www.acmicpc.net/problem/15649

오늘의 메모(선택): 백트래킹 문제입니다. 재귀에서 return 하는 곳, 배열의 각 칸에 수를 넣기 위해 isSelected 인지 체크하는 부분 주의할 것 !!

코드(선택): cpp

```cpp
#include <iostream>
#include <vector>
using namespace std;

int n, m;

void solution(vector<int> &arr, vector<bool> &isSelected, int cnt) {
	if (cnt == m) {
		for (int i = 0; i < m; i++) {
			cout << arr[i] << " ";
		}
		cout << "\n";
		return ;
	}
	for (int i = 1; i <= n; i++) {
		if (!isSelected[i]) {
			isSelected[i] = true;
			arr[cnt] = i;
			solution(arr, isSelected, cnt + 1);
			isSelected[i] = false;
		}
	}
}

int main() {
	ios::sync_with_stdio(false);
	cin.tie(NULL);
	cout.tie(NULL);
	
	cin >> n >> m;
	vector<int> arr(m, 0);
	vector<bool> isSelected(n + 1, false);
	solution(arr, isSelected, 0);
	return 0;
}
```
