#include <stdio.h>

void ch(int *stk, int last)
{
        printf("in stack %d :", last);
    for (int i = 0; i < last; i++)
    {
        printf("%d %d / ", i, stk[i]);
    }
    printf("\n");
}
// arr_len은 배열 arr의 길이입니다.
int* solution(int arr[], size_t size) {
    // return 값은 malloc 등 동적 할당을 사용해주세요. 할당 길이는 상황에 맞게 변경해주세요.
    int* stk = (int*)malloc(sizeof(int) * size);
    int i = 0, last = 0;;
    stk[last] = 0;

    while (i < size)
    {        
        if (last == 0)
        {
            stk[last] = arr[i];
            ++i;
            ++last;
            // ch(stk,last)   ;
        }
        else
        {            
            if (stk[last -1] < arr[i])
            {
                stk[last] = arr[i];
                last++;
                i++;                
                // ch(stk,last)   ;
            }
            else
            {
                stk[last] = 0;
                last--;
                // ch(stk,last)   ;
            }
        }        
    }
    return stk;
}
