**코드(선택)**:

```c
#include <stdio.h>
#include <stdbool.h>
#include <stdlib.h>


int cmp (const int *a, const int *b)
{
    return (*b - *a);
}

int* solution(int k, int score[], size_t score_len) {
    int* answer = calloc(score_len, sizeof(int));
    int* tmp = calloc(k, sizeof(int));
    int i = 0, last = 0;
    
    while (i < score_len)
    {
        if(i < k)
        {
            tmp[i] = score[i];
            last = i;
        }
        else
        {
             if (tmp[last] <= score[i])
                tmp[last] = score[i];
        }
        qsort(tmp, k, sizeof(int), cmp);
        answer[i] = tmp[last];
        i++;
    }
    free(tmp);
    
    return answer;
}

```
