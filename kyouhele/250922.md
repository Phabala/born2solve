#include <stdio.h>
#include <stdlib.h>

int cmp(const void* a, const void* b) {
    return (*(int*)a - *(int*)b);
}

int* solution(int num_list[], size_t num_list_len) {
    qsort(num_list, num_list_len, sizeof(int), cmp); // 정렬
    int* answer = malloc(5 * sizeof(int));
    for (int i = 0; i < 5; i++) {
        answer[i] = num_list[i]; // 앞에서 5개
    }
    return answer;
}
