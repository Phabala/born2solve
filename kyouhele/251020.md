#include <stdio.h>
#include <stdbool.h>
#include <stdlib.h>
#include <math.h>

int get_len(int x, int y, int dist)
{
    long long len = ((long long)x*x) + ((long long) y*y);
    if (len <= (long long)dist * dist)
        return 1;
    else
        return 0;
}

long long solution(int k, int d) {
    long long cnt = 0;
    for (long long  x = 0; x <= d; x += k)
    {
        long long maxY = sqrt(((long long)d * d) - (x * x));
        cnt += maxY / k + 1;
    }
    return cnt;    
}
