**코드(선택)**:

```c
#include <stdio.h>
#include <stdbool.h>
#include <stdlib.h>

long long solution(int n) {
    if (n == 1) return 1;
    if (n == 2) return 2;
    int *dp = (int *) malloc (sizeof(int) * (n + 1));
    dp[0] = 0;
    dp[1] = 1;
    dp[2] = 2;
    
    for (int i = 3; i <= n; i++)
        dp[i] = (dp[i - 1] + dp[i - 2]) % 1234567;
    
    return dp[n];
}
```
