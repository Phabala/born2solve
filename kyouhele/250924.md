**코드(선택)**:

```c

#include <stdio.h>
#include <stdbool.h>
#include <stdlib.h>

int solution(int** arr, size_t arr_rows, size_t arr_cols) {
    int answer = 0,cnt = 0;
    
    for (int i = 0; i < arr_rows; i++)
    {
        for (int j = 0; j <  arr_cols; j++)
        {
            if (arr[i][j] == arr[j][i])
                cnt++;
        }
    }
    if (cnt == arr_rows * arr_cols)        answer = 1;
    return answer;
}
```
