**코드(선택)**:

```c
#include <stdio.h>
#include <stdbool.h>
#include <stdlib.h>

int solution(int num_list[], size_t num_list_len) {
    int cnt = 0;
    for (int i = 0; i < num_list_len; i++)
    {
        for (int num = num_list[i]; num > 1; num /= 2)
        {
            if (num % 2 == 1)
                num -= 1;
            cnt++;
        }
    }
    
    return cnt;
}
```
