```C
#include <stdio.h>
#include <stdbool.h>
#include <stdlib.h>

char* solution(const char* str[], size_t str_len, int** p, size_t p_rows, size_t p_cols) {
    int size = 0, j = 0;
    for (int i = 0; i < p_rows; ++i)
        size += p[i][1] - p[i][0] +1;
    char* ret = (char*)malloc(size + 1);
    
    for (int row = 0; row < p_rows; row++)
    {
        for (int i = p[row][0]; i <= p[row][1]; ++i)
            ret[j++] = str[row][i];
    }
    ret[j] = 0;
    return ret;
}
```
