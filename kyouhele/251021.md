//1부터n까지 n(100 이하)명 player idx & 스코어 주어짐  스코어로 순위로 sort 그런데 몇 개 실종
// [4,3] -> 4번이 3번 이김 [1,4] -> 1번이 4번 이김



#include <iostream>
#include <vector>
using namespace std;

int solution(int n, vector<vector<int>> results) {
    vector<vector<int>> score(n + 1, vector<int>(n + 1, 0));

    for (auto &r : results) {
        int win = r[0];
        int lose = r[1];
        score[win][lose] = 1;
    }

    for (int k = 1; k <= n; k++) {
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= n; j++) {
                if (score[i][k] && score[k][j])
                    score[i][j] = 1;
            }
        }
    }

    int answer = 0;
    for (int i = 1; i <= n; i++) {
        int cnt = 0;
        for (int j = 1; j <= n; j++) {
            if (score[i][j] || score[j][i])
                cnt++;
        }
        if (cnt == n - 1)
            answer++;
    }

    return answer;
}
