**코드(선택)**:

```c
#include <stdio.h>
#include <stdbool.h>
#include <stdlib.h>

char* solution(int q, int r, const char* code) {
    int i = 0, len = 0, idx = 0;
    while(code[len])
        len++;
    
    char* ret = (char*)malloc(len / q + 1);
    while(i * q + r < len)
    {
        ret[idx]  = code[i * q + r];
        idx++;
        i++;
    }
    ret[i] = 0;
    
    return ret;
}
```
