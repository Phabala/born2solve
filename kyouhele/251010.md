**코드(선택)**:

```c
#include <stdio.h>
#include <stdbool.h>
#include <stdlib.h>

int solution(int hp) {
    int ret = 0, point = hp;
    
    for (point; point > 0; ret++)
    {
        if (point >= 5)
            point -=5;
        else if (point >= 3)
            point -= 3;
        else
            point--;
    }
    return ret;
}
```
