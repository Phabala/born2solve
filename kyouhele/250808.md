**오늘의 메모(선택)**: 결국 c로 해결했다 배열만들기2

**코드(선택)**:

```C

#include <stdio.h>
#include <stdbool.h>
#include <stdlib.h>

int* solution(int l, int r) {
    int* ret = calloc(r, sizeof(int));
    int i = l, val, tmp, digit, j = 0;
    while (i <= r)
    {
        tmp = i, val = 0, digit = 1;
        
        while (tmp >= 5)
        {
            if (tmp % 10 == 5 || tmp % 10 == 0)
                val = val + (tmp%10 * digit);
            else
            {
                val  = -1;
                break;
            }
            tmp = tmp / 10;
            if (tmp % 10 != 5 && tmp % 10 != 0)
            {
                val  = -1;
                break;
            }                
            digit *= 10;
        }
        if (val > 0)
            *(ret+j++) = i;
        i++;
    }
    if (j == 0)
        ret[0] = -1;
    return ret;
}
```
