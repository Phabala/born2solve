**코드(선택)**:

```c
#include <stdio.h>
#include <stdbool.h>
#include <stdlib.h>

char** solution(const char* str[], size_t str_len) {
    int size = 0,idx = 0,i = 0, start=0;
    while(idx < str_len)
    {
        if (str[idx][0] == 'l')
        {
            size = idx;
            break;
        }
        else if (str[idx][0] == 'r')
        {
            size = str_len  - idx + 1;
            break;
        }
        idx++;
    }
    
    char** answer = (char**)malloc(sizeof(char*) * (size + 1));
    if (str[idx][0] == 'r')
        start = idx + 1;
    for (i;  i < size; i++)
    {
        answer[i] = str[start];
        start++;
    }
    answer[size] = NULL;
    
    return answer;
}
```
