**코드(선택)**:

```c
#include <stdio.h>
#include <stdbool.h>
#include <stdlib.h>

// topping_len은 배열 topping의 길이입니다.
int solution(int topping[], size_t topping_len) {
    int left[100001]={0};
    int right[100001]={0};
    int same = 0, cnt_left = 0,cnt_right=0;
    
    for (int i = 0; i < topping_len; i++)
    {
        if (right[topping[i]] == 0)
            cnt_right++;
        right[topping[i]] ++;
    }
    for (int i = 0; i < topping_len - 1; i++)
    {
        int j = topping[i];
        
        if (left[j] == 0)
            cnt_left++;
        left[j] ++;
        
        right[j]--;
        if (right[j] == 0)
            cnt_right--;
        
        if (cnt_left == cnt_right)
            same++;
    }
    
    return same;
}
```
