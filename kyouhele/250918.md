**코드(선택)**:

```c
#include <stdio.h>
#include <stdbool.h>
#include <stdlib.h>
#include <string.h>

char* solution(const char* rny) {
    int size = strlen(rny), j = 0;
    for (int i = 0; rny[i]; i++)
    {
        if (rny[i] == 'm')
            size ++;
    }
    char* answer = (char*)malloc(size + 1);
    for (int i = 0; i < size; i++)
    {
        answer[i] = rny[j];
        if (rny[j] == 'm')
        {
            answer[i] = 'r';
            answer[++i] = 'n';
        }
        j++;
    }
    answer[size]= 0;
    return answer;
}

```
