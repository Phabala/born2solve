**오늘의 메모(선택)**:

누적 합 2차원 버전 문제

	ios::sync_with_stdio(false);
	cin.tie(0);

입출력이 많은 경우 위 두 줄 추가하면 미세하게 빨라지는데 외우기 귀찮아서 일단 string 에 답 쌓았다

**코드(선택)**:

```c++
#include <iostream>
#include <string>

using namespace std;

int N, M;

int main() {
	// input & init
	cin >> N >> M;
	int prefixSum[N + 1][N + 1];

	for (int i = 0; i <= N; i++) {
		prefixSum[0][i] = 0;
		prefixSum[i][0] = 0;
	}

	// solve
	for (int row = 1; row <= N; row++) {
		for (int col = 1; col <= N; col++) {
			cin >> prefixSum[row][col];
			prefixSum[row][col] = (
				prefixSum[row][col]
				+ prefixSum[row - 1][col]
				+ prefixSum[row][col - 1]
				- prefixSum[row - 1][col - 1]
			);
		}
	}

	// solve
	int x1, y1, x2, y2;
	string ans;
	for (int i = 0; i < M; i++) {
		cin >> x1 >> y1 >> x2 >> y2;
		ans += to_string(
			prefixSum[x2][y2]
			- prefixSum[x2][y1 - 1]
			- prefixSum[x1 - 1][y2]
			+ prefixSum[x1 - 1][y1 - 1]
		) += '\n';
	}

	// output
	cout << ans;
}
```
