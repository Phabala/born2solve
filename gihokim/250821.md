**오늘의 메모(선택)**: 

**코드(선택)**:

```c++
#include <iostream>
#include <string>
#include <unordered_map>

using namespace std;

int N, M;
unordered_map<string, string> map; // key: URL, value: PW

int main() {
	// input
	cin >> N >> M;

	string url, pw;
	for (int i = 0; i < N; i++) {
		cin >> url >> pw;
		map.insert({url, pw});
	}

	// solve & output
	string ans = "";
	for (int i = 0; i < M; i++) {
		cin >> url;
		ans += (map.find(url))->second;
		ans += '\n';
	}
	
	cout << ans;
}
```
