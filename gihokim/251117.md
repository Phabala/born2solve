**문제 이름 또는 번호**: 1437. Check If All 1's Are at Least Length K Places Away

**문제 링크**: https://leetcode.com/problems/check-if-all-1s-are-at-least-length-k-places-away/

**오늘의 메모(선택)**:

**코드(선택)**:

```c++
#include <vector>
#include <algorithm>

class Solution {
public:
    bool kLengthApart(vector<int>& nums, int k) {
        vector<int> oneIndexes;

        for (int i = 0; i < nums.size(); i++) {
            if (nums[i] == 1)
                oneIndexes.push_back(i);
        }

        if (oneIndexes.size() < 2)
            return true;

        int minLen = 21e8;
        for (int i = 0; i < oneIndexes.size() - 1; i++) {
            int j = i + 1;
            minLen = min(oneIndexes[j] - oneIndexes[i] - 1, minLen);
            if (minLen < k)
                return false;
        }
        return true;
    }
};
```
