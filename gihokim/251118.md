**문제 이름 또는 번호**: 717. 1-bit and 2-bit Characters (실패)

**문제 링크**: https://leetcode.com/problems/1-bit-and-2-bit-characters

**오늘의 메모(선택)**: DP 같긴 한데 아직 잘 모르겠다. 아래는 실패하는 코드

**코드(선택)**:

```c++
#include <iostream>

class Solution {
private:
    bool isTrue(vector<int>& bits, int i, vector<bool> memo) {
        if (memo[i])
            return true;
        if (i == 1) {
            if (bits[0] == 1) {
                memo[0] = true;
                memo[1] = true;
                return true;
            } else {
                return false;
            }
        }
        if (memo[i - 1] && bits[i] == 0)
            return memo[i] = true;
        if (memo[i - 2] && bits[i - 1] == 1) {
            memo[i - 1] = true;
            memo[i] = true;
            return true;
        }
        return isTrue(bits, i - 1, memo);
    }
public:
    bool isOneBitCharacter(vector<int>& bits) {
        if (bits.size() == 1)
            return false;
        bits.pop_back();
        vector<bool> memo(bits.size(), false);
        return isTrue(bits, bits.size() - 1, memo);
    }
};
```
