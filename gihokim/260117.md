**오늘의 메모(선택)**: 알고리즘 재활 용으로 좋았던 문제

**코드(선택)**:

```c++
#include <iostream>
#include <algorithm>
#include <vector>
#include <set>

using namespace std;

class Node {
public:
	string str;
	size_t len;

	Node(string str) : str(str), len(str.length()) {}

	bool operator< (const Node other) const {
		if (len < other.len)
			return true;
		else if (len > other.len)
			return false;

		if (str < other.str)
			return true;
		else if (str > other.str)
			return false;

		return false;
	}
};

int main() {
    int N;
    cin >> N;
    
    set<string> s;
    
    for (int i = 0; i < N; i++) {
        string str;
        cin >> str;
        s.insert(str);
    }
    
    vector<Node> v;
    for (auto it = s.begin(); it != s.end(); it++) {
        v.push_back(Node(*it));
    }
    
    sort(v.begin(), v.end());
    
    for (int i = 0; i < v.size(); i++)
        cout << v[i].str << '\n';
}
```
