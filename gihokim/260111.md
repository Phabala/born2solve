**오늘의 메모(선택)**: 

**코드(선택)**:

```c++
#include <iostream>

using namespace std;

int T;

int main() {
	cin >> T;

	for (int tc = 1; tc <= T; tc++) {
		int N;
		cin >> N;
		for (int i = 0; i < N; i++) {
			for (int j = 0; j < N; j++) {
				if (i == 0 || i == N - 1 || j == 0 || j == N - 1)
					cout << '#';
				else
					cout << 'J';
			}
			cout << '\n';
		}
		cout << '\n';
	}
}
```
