**오늘의 메모(선택)**: 

그냥 map 에 있는 값 다 더하면 시간초과 될 거기 때문에 구간별로 나눠서 합하려고 했는데 디버깅이 너무 어려웠다.
그리고 이렇게 푸는게 아니라 DP로 풀어야 한다고 한다.

**코드(선택)**:

```c++
#include <iostream>
#include <vector>

using namespace std;

#define BLOCK 25
int N, M;
vector<vector<int>> map;
vector<vector<int>> cmd;
vector<vector<int>> mosaic;

void init_mosaic() {
	int y = (N / BLOCK) + 1;
	int x = (N / BLOCK) + 1;
	mosaic = vector<vector<int>>(y, vector<int>(x, 0));

	for (int offset_i = 0; offset_i < y; offset_i++) {
		for (int offset_j = 0; offset_j < x; offset_j++) {
			for (int i = 0; i < BLOCK && offset_i * BLOCK + i < N; i++) {
				for (int j = 0; j < BLOCK && offset_j * BLOCK + j < N; j++) {
					mosaic[offset_i][offset_j] += map[offset_i * BLOCK + i][offset_j * BLOCK + j];
				}
			}
		}
	}

	for (int i = 0; i < mosaic.size(); i++) {
		for (int j = 0; j < mosaic[i].size(); j++) {
			cout << mosaic[i][j] << ' ';
		}
		cout << '\n';
	}
}

int calc(int x1, int y1, int x2, int y2) {
	int quotient_i = (x2 - x1) / BLOCK;
	int quotient_j = (y2 - y1) / BLOCK;
	int remainder_i = (x2 - x1) % BLOCK;
	int remainder_j = (y2 - y1) % BLOCK;
	int ans = 0;

	for (int i = 0; i < quotient_i; i++) {
		for (int j = 0; j < quotient_j; j++) {
			ans += mosaic[i][j];
		}
	}

	for (int i = 0; i <= remainder_i; i++) {
		for (int j = 0; j <= remainder_j; j++) {
			ans += map[quotient_i * BLOCK + i][quotient_j * BLOCK + j];

		}
	}

	return ans;
}

int main() {
	// input & init
	cin >> N >> M;
	map = vector<vector<int>>(N);
	cmd = vector<vector<int>>(N);

	int num;
	for (int i = 0; i < N; i++) {
		for (int j = 0; j < N; j++) {
			cin >> num;
			map[i].push_back(num);
		}
	}

	for (int i = 0; i < M; i++) {
		for (int j = 0; j < 4; j++) {
			cin >> num;
			cmd[i].push_back(num - 1);
		}
	}

	// solve & output
	init_mosaic();
	for (int tc = 0; tc < M; tc++) {
		int x1, y1, x2, y2;
		x1 = cmd[tc][0];
		y1 = cmd[tc][1];
		x2 = cmd[tc][2];
		y2 = cmd[tc][3];
cout << "DEBUG: tc: " << tc << ", x1 y1 x2 y2: " << x1 << ' ' << y1 << ' ' << x2 << ' ' << y2 << '\n';
		cout << calc(x1, y1, x2, y2) << '\n';
	}
}
```
