**문제 이름 또는 번호**: 21. Merge Two Sorted Lists

**문제 링크**: https://leetcode.com/problems/merge-two-sorted-lists/

**오늘의 메모(선택)**: 원래 있는 노드 재활용 안하고 시원하게 링크드리스트 하나 더 만들었다.

**코드(선택)**:

```c++
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
#include <iostream>
#include <algorithm>

using namespace std;

class Solution {
public:
    ListNode* mergeTwoLists(ListNode* list1, ListNode* list2) {
        vector<int> v;

        while (list1) {
            v.push_back(list1->val);
            list1 = list1->next;
        }
        while (list2) {
            v.push_back(list2->val);
            list2 = list2->next;
        }

        sort(v.begin(), v.end());

        ListNode* head = NULL;
        ListNode* tmp = NULL;
        if (v.size() > 0) {
            head = new ListNode(v[0]);
            tmp = head;
        }
        for (int i = 1; i < v.size(); i++) {
            tmp->next = new ListNode(v[i]);
            tmp = tmp->next;
        }
        return head;
    }
};
```
