**문제 이름 또는 번호**: 3318. Find X-Sum of All K-Long Subarrays I (실패)

**문제 링크**: https://leetcode.com/problems/find-x-sum-of-all-k-long-subarrays-i

**오늘의 메모(선택)**: 요즘 쉬운것만 풀었더니 구현 난이도 살짝만 올라간 easy 문제인데 버벅대는중

**코드(선택)**:

```c++
class Solution {
private:
    int dict[51] = {0}; // key: n, value: count

    void init(int i1, int i2, vector<int>& nums) {
        for (int j = i1; j <= i2; j++)
            dict[nums[j]]++;
    }

    void update(int i1, int i2, vector<int>& nums) {
        dict[nums[i1 - 1]]--;
        dict[nums[i2]]++;
    }

    int getAnswer(int i1, int i2) {
        int max1st = 0;
        int max2nd = 0;

        for (int j = 0; j < 51; j++) {
            if (dict[j] >= dict[max1st]) {
                max2nd = max1st;
                max1st = j;
            } else if (dict[j] >= dict[max2nd]) {
                dict[max2nd] = j;
            }
        }
cout << "DEBUG: max1st: " << max1st << ", max2nd: " << max2nd << '\n';
        return max1st * dict[max1st] + max2nd * dict[max2nd];
    }
    
public:
    vector<int> findXSum(vector<int>& nums, int k, int x) {
        int i1 = 0;
        int i2 = k - 1;
        vector<int> ans;

        init(i1, i2, nums);

        while (i2 < nums.size()) {
            ans.push_back(getAnswer(i1, i2));
            i1++;
            i2++;
            if (i2 < nums.size())
                update(i1, i2, nums);
        }

        return ans;
    }
};
```
