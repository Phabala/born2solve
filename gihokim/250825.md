**오늘의 메모(선택)**: 

실버3 치고는 어려웠다.

내가 한 방법은 2^N 이여서 시간초과가 나서, 수학적 조합론으로 접근해야 한다고 한다.

**코드(선택)**:

> 내가 푼 코드

```c++
#include <iostream>
#include <map>
#include <vector>

using namespace std;

int T, N;
map<string, int> m;
unsigned long long ans;
string str_ans;

void dfs(vector<int> cnt, int i, unsigned long long qty) {
    ans += qty;
    if (i == cnt.size() - 1)
        return;
    for (int j = i + 1; j < cnt.size(); j++)
        dfs(cnt, j, qty * cnt[j]);
}

int main() {
    cin >> T;
    for (int tc = 1; tc <= T; tc++) {
        // reset
        m.clear();
        ans = 0;

        // input
        cin >> N;
        string key, value;
        for (int i = 0; i < N; i++) {
            cin >> value >> key;
            if (m.find(key) == m.end()) m.insert({key, 1});
            else m.find(key)->second++;
        }
        
        // solve
        vector<int> cnt;
        for (auto it = m.begin(); it != m.end(); it++)
            cnt.push_back(it->second);
        for (int i = 0; i < cnt.size(); i++)
            dfs(cnt, i, cnt[i]);

        // output
        str_ans += to_string(ans) += '\n';
    }

    cout << str_ans;
}
```

> 수학적 조합론

```c++
#include <iostream>
#include <map>
#include <string>

using namespace std;

int main() {
    int T;
    cin >> T;
    
    while (T--) {
        int N;
        cin >> N;
        
        map<string, int> clothes;
        
        for (int i = 0; i < N; i++) {
            string name, type;
            cin >> name >> type;
            clothes[type]++;
        }
        
        int result = 1;
        for (auto& pair : clothes) {
            // 각 종류마다 (해당 종류 의상 수 + 1)을 곱함
            // +1은 그 종류를 아예 입지 않는 경우
            result *= (pair.second + 1);
        }
        
        // 알몸인 경우(모든 종류를 입지 않는 경우) 제외
        result--;
        
        cout << result << "\n";
    }
    
    return 0;
}
```
