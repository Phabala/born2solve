**문제 이름 또는 번호**: 3318. Find X-Sum of All K-Long Subarrays I

**문제 링크**: https://leetcode.com/problems/find-x-sum-of-all-k-long-subarrays-i/

**오늘의 메모(선택)**: unordered_map 을 while 문 안에서 맨날 세팅하는게 아니라 슬라이딩 윈도우 이용하면 시간복잡도 줄일 수 있는데 그냥 함

**코드(선택)**:

```c++
#include <unordered_map>
#include <utility>
#include <algorithm>

class Solution {
private:
    static bool isMoreFrequent(pair<int, int> a, pair<int, int> b) {
        return (a.second > b.second || (a.second == b.second && a.first > b.first));
    }
public:
    vector<int> findXSum(vector<int>& nums, int k, int x) {
        int i1 = 0;
        int i2 = k - 1;

        vector<int> ret;
        while (i2 < nums.size()) {
            unordered_map<int, int> um; // key: num, value: cnt
            for (int i = i1; i <= i2; i++) {
                if (um.find(nums[i]) == um.end())
                    um[nums[i]] = 1;
                else
                    um.find(nums[i])->second++;
            }

            vector<pair<int, int>> freq;
            for (auto it = um.begin(); it != um.end(); it++)
                freq.push_back(*it);
            sort(freq.begin(), freq.end(), isMoreFrequent);
            int ans = 0;
            for (int i = 0; i < min((int)freq.size(), x); i++)
                ans += freq[i].first * freq[i].second;
            ret.push_back(ans);
            i1++;
            i2++;
        }
        return ret;
    }
};
```
