**문제 이름 또는 번호**: 1267 / 핸드폰 요금

**문제 링크**: https://www.acmicpc.net/problem/1267

**오늘의 메모(선택)**: 브론즈. 인데 아직 적절한 수식을 못찾음

**코드(선택)**:

```c++
#include <iostream>
#include <algorithm>

using namespace std;

int main() {
	int N;
	cin >> N;
	
	int y = 0;
	int m = 0;

	for (int i = 0; i < N; i++) {
		int minute;
		cin >> minute;

		y += (10 * (minute / 30)) + (10 * !!(minute / 30)) + (10 * !!(minute % 30));
		m += (15 * (minute / 60)) + (15 * !!(minute / 60)) + (15 * !!(minute % 60));
		cout << "DEBUG: " << (10 * (minute / 30)) + (10 * !!(minute / 30)) + (10 * !!(minute % 30)) << ' ' << (15 * (minute / 60)) + (15 * !!(minute / 60)) + (15 * !!(minute % 60)) << '\n';
	}

	if (y < m)
		cout << "Y ";
	else if (y > m)
		cout << "M ";
	else
		cout << "Y M ";

	cout << min(y, m);
}
```
