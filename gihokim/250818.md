**코드(선택)**:

```c++
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
	int N;
	cin >> N;
	vector<int> row;

	int num;
	for (int i = 0; i < N; i++) {
		cin >> num;
		row.push_back(num);
	}

	sort(row.begin(), row.end());

	unsigned long long ans = 0;
	for (int i = 0; i < N; i++)
		ans += row[i] * (row.size() - i);
	
	cout << ans;
}

```
