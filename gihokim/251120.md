**오늘의 메모(선택)**: 브론즈.

**코드(선택)**:

```c++
#include <iostream>

using namespace std;

void swap(int table[3], int x, int y) {
	int xi = 0;
	int yi = 0;

	while (table[xi] != x)
		xi++;
	while (table[yi] != y)
		yi++;
	table[yi] = x;
	table[xi] = y;
}

int main() {
	int cnt;
	cin >> cnt;

	int table[3] = {1, 2, 3};
	for (int c = 0; c < cnt; c++) {
		int x, y;
		cin >> x >> y;
		swap(table, x, y);
	}

	cout << table[0];
}
```
