**오늘의 메모(선택)**: 

원형 구조를 선형 DP 두 번으로 나눠 계산해 최댓값을 선택한다.

**코드(선택)**: 

```c++
#include <vector>
#include <algorithm>

using namespace std;

int solution(vector<int> money) {
  int n = money.size();
  vector<int> dp1(n, 0);
  vector<int> dp2(n, 0);

  dp1[0] = money[0];
  dp1[1] = money[0];
  for (int i = 2; i < n - 1; ++i) {
    dp1[i] = max(dp1[i - 1], dp1[i - 2] + money[i]);
  }

  dp2[1] = money[1];
  for (int i = 2; i < n; ++i) {
    dp2[i] = max(dp2[i - 1], dp2[i - 2] + money[i]);
  }

  int answer = max(dp1[n - 2], dp2[n - 1]);
  return answer;
}

```
