**코드(선택)**:
```cpp

#include <iostream>
#include <set>

int	main()
{
	std::ios::sync_with_stdio(0);
	std::cin.tie(0);

	std::string	cmd;
	int	M, number;
	std::set<int>	numbers;

	std::cin >> M;
	while (M--)
	{
		std::cin >> cmd;

		if (cmd == "add")
		{
			std::cin >> number;
			numbers.insert(number);
		}
		else if (cmd == "remove")
		{
			std::cin >> number;
			numbers.erase(number);
		}
		else if (cmd == "check")
		{
			std::cin >> number;
			if (numbers.count(number) > 0)
				std::cout << "1\n";
			else
				std::cout << "0\n";
		}
		else if (cmd == "toggle")
		{
			std::cin >> number;
			if (numbers.count(number) > 0)
				numbers.erase(number);
			else
				numbers.insert(number);
		}
		else if (cmd == "all")
		{
			for (int i = 1; i <= 20; ++i)
				numbers.insert(i);
		}
		else if (cmd == "empty")
			numbers.clear();
	}

	return 0;
}
```
