**오늘의 메모(선택)**: 아니 무슨 초기에 5번째까지 세팅해주는 수열이 어딨냐고 그렇게 만들어놓고 아 무슨 수열이.. 그럴리가 없어 라면서 혼자 규칙찾다가 시간 다 태웠다.

**코드(선택)**:

```c++
#include <iostream>
#include <vector>

using namespace std;

long long wave(int num) {
	vector<long long> v(101);
	v[1] = 1;
	v[2] = 1;
	v[3] = 1;
	v[4] = 2;
	v[5] = 2;
	if (num <= 5) {
		return v[num];
	}
	for (int i = 6; i <= num; i++) {
		v[i] = v[i - 1] + v[i - 5];
	}
	return v[num];
}
int main() {
	int T, N;
	cin >> T;
	for (int i = 0; i < T; i++) {
		cin >> N;
		cout << wave(N) << '\n';
	}
	return (0);
}
```
