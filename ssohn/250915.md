**오늘의 메모(선택)**: emplace_back()이라는 함수에 대해서 공부했다

참고 :

**코드(선택)**:

```c++
#include <iostream>
#include <deque>

using namespace std;

struct s_dq
{
	int idx;
	int num;

	s_dq(int i, int n) : idx(i), num(n) {}
};


int main() {
	int N;
	int idx = 1;
	int num;
	cin >> N;
	deque<s_dq> dq;  
	for (int i = 0; i < N; i++) {
		cin >> num;
		dq.emplace_back(idx, num);
		idx++;
	}

	while(!dq.empty()) {
		cout << dq.front().idx << ' ';
		int temp = dq.front().num;
		dq.pop_front();

		if (dq.empty())
			break;
		
		if (temp > 0) {
			for (int i = 0; i < temp - 1; i++) {
				dq.push_back(dq.front());
				dq.pop_front();
			}
		} else {
			for (int i = 0; i < -temp; i++) {
				dq.push_front(dq.back());
				dq.pop_back();
			}
		}
	}

	return (0);
}
```
