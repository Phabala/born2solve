**오늘의 메모(선택)**: vector의 find의 경우 선형 탐색을 하는 반면 set의 경우 이진트리 탐색을 하기 때문에 시간이 훨씨 빠르다는 것을 알았다.

참고 :

**코드(선택)**:

```c++
#include <iostream>
#include <set>
#include <algorithm>

using namespace std;

int main() {
	ios::sync_with_stdio(false);
	cin.tie(NULL);
	int N;
	int count = 0;
	string name;
	set<string> s;
	
	cin >> N;
	for (int i = 0; i < N; i++) {
		cin >> name;
		if (name == "ENTER") {
			s.clear();
		} else if (s.find(name) == s.end()) {
			s.insert(name);
			count++;
		}
	}
	cout << count << '\n';
	return (0);
}
```
