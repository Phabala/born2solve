```Go

import "sort"

func solution(n int64) int64 {
	var digits []int64

	for n > 0 {
		digits = append(digits, n%10)
		n /= 10
	}

	sort.Slice(digits, func(i, j int) bool {
		return digits[i] > digits[j]
	})

	var result int64
	for _, d := range digits {
		result = result*10 + d
	}

	return result
}
