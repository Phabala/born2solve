#include <string>
#include <vector>

using namespace std;

vector<int> solution(int n) {
    vector<int> answer;
    vector<vector<int>> triangle(n);
    
    for (int i = 0; i < n; i++)
        triangle[i].resize(i + 1, 0);

    int x = -1, y = 0;
    int num = 1;

    for (int d = 0; d < n; d++) {
        for (int i = d; i < n; i++) {
            if (d % 3 == 0) {
                x++;
            }
            else if (d % 3 == 1) {
                y++;
            }
            else {
                x--;
                y--;
            }
            triangle[x][y] = num++;
        }
    }

    for (auto row : triangle)
        for (int val : row)
            answer.push_back(val);
    
    return answer;
}
