#include <string>
#include <vector>
#include <climits>
#include <algorithm>

using namespace std;


void cal_weigth(vector<vector<int>> &w, vector<vector<int>> &fares) {
    int n = fares.size();
    int m = fares[0].size();
    for (int i = 0; i < n; i++) {
        int u = fares[i][0]-1;
        int v = fares[i][1]-1;
        int cost = fares[i][2];
        w[u][v] = cost;
        w[v][u] = cost;
    }
}

void Dijkstra(int s, vector<vector<int>> &w, vector<int> &d) {
    int n = w.size();
    d[s] = 0;
    vector<bool> visit(n);
    
    for (int i = 0; i < n; ++i) {
        int u = -1;
        int min_dist = INT_MAX;
        for (int j = 0; j < n; ++j) {
            if (!visit[j] && d[j] < min_dist) {
                min_dist = d[j];
                u = j;
            }
        }

        if (u == -1)
            break;
        visit[u] = true;
        for (int v = 0; v < n; ++v) {
            if (w[u][v] > 0 && !visit[v]) {
                if (d[u] + w[u][v] < d[v]) {
                    d[v] = d[u] + w[u][v];
                }
            }
        }
    }
}

int solution(int n, int s, int a, int b, vector<vector<int>> fares) {
    int answer = INT_MAX;
    vector<vector<int>> w(n, vector<int>(n, 0));
    vector<int> ds(n, INT_MAX);
    vector<int> da(n, INT_MAX);
    vector<int> db(n, INT_MAX);
    
    cal_weigth(w,fares);
    
    Dijkstra(s-1,w,ds);
    Dijkstra(a-1,w,da);
    Dijkstra(b-1,w,db);
    
    for (int k = 0; k < n; ++k) {
        if (ds[k] != INT_MAX && da[k] != INT_MAX && db[k] != INT_MAX) {
            answer = min(answer, ds[k] + da[k] + db[k]);
        }
    }
        
    return answer;
}
