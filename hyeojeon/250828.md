#include <string>
#include <vector>
#include <algorithm>
#include <sstream>

using namespace std;

bool check(int num, vector<int> &answer) {
    return find(answer.begin(), answer.end(), num) == answer.end();
}

vector<int> solution(string s) {
    vector<int> answer;
    vector<string> tuples;
    int n = s.size();
    
    for (int i = 1; i < n-2; i++) {
        if (s[i] == '{') {
            i++;
            string str;
            
            while (s[i] != '}')
                str+=s[i++];
            
            tuples.push_back(str);
        }
    }
    
    std::sort(tuples.begin(), tuples.end(), [](const std::string& a, const std::string& b) {
        return a.length() < b.length();
    });
    
    vector<vector<int>> res;
    for (const auto& str : tuples) {
        vector<int> nums;
        stringstream ss(str);
        string token;

        while (getline(ss, token, ',')) {
            nums.push_back(stoi(token));
        }

        res.push_back(nums);
    }
    
    answer.push_back(res[0][0]);
    
    for (int i = 1; i < res.size(); i++) {
        for (int j = 0; j < res[i].size(); j++) {
            if (check(res[i][j], answer))
                answer.push_back(res[i][j]);
        }
    }
    
    return answer;
}
