#include <string>
#include <vector>
#include <set>

using namespace std;

int solution(vector<int> elements) {
    int answer = 0;
    int n = elements.size();
    vector<int> vec(2*n,0);
    for (int i = 0 ; i < 2*n; i++)
        vec[i] = elements[i%n];
    set<int> res;
    int k = 1;
    int sum = 0;
    for (; k <= n; k++) {
        for (int i = 0; i < n; i++) {
            sum = 0;
            for (int j = 0; j < k; j++) {
                sum += vec[i+j];
            }
            res.insert(sum);
        }
    }
    return res.size();
}
