```cpp

#include <iostream>
using namespace std;

int solution(int n)
{
    int ans = 0;
    int i = 0;
    while (i < sizeof(int) * 8) {
        if ((n >> i) & 1)
            ans++;
        i++;
    }
    return ans;
}

```
