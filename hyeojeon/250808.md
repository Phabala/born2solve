#include <string>
#include <vector>
#include <algorithm>

using namespace std;

vector<int> solution(int n, vector<string> words) {
    vector<int> answer;

    int len = words.size();
    string prv = words[0];
    int i = 1;
    for (; i < len; i++) {
        string now = words[i];
        if (now[0] != prv[prv.size()-1] || find(words.begin(), words.begin() + i, now) != words.begin() + i)
            break;
        prv = words[i];
    }
    if (i == len)
        answer = {0,0};
    else {
        answer.push_back((i%n)+1);
        answer.push_back((i/n)+1);
    }
    return answer;
}
