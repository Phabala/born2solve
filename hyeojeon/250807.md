```cpp

#include <string>
#include <vector>

using namespace std;

long long cnt(int n) {
    if (n == 1)
        return 1;
    if (n == 2)
        return 2;
    vector<int> dp(n, 0);
    dp[0] = 1;
    dp[1] = 2;
    for (int i = 2; i < n; i++)
        dp[i] = (dp[i-1] + dp[i-2]) % 1234567;
    return dp[n-1];
}

long long solution(int n) {
    long long answer = 0;
    answer = cnt(n) % 1234567;
    return answer;
}

```
