#include <string>
#include <vector>
#include <stack>

using namespace std;

string solution(string number, int k) {
    stack<char> st;
    
    for (char c : number) {
        while (!st.empty() && k > 0 && c > st.top()) {
            st.pop();
            k--;
        }
        st.push(c);
    }
    
    while (k > 0 && !st.empty()) {
        st.pop();
        k--;
    }
    
    string res(st.size(), '0');
    for (int i = st.size() - 1; i >= 0; i--) {
        res[i] = st.top();
        st.pop();
    }
    
    return res;
}
