**오늘의 메모(선택)**: set 최고

**코드(선택)**:

```cpp

#include <string>
#include <vector>
#include <algorithm>
#include <set>

using namespace std;

int solution(vector<int> scoville, int K) {
    int answer = 0;
    multiset<int> ms(scoville.begin(), scoville.end());
    while (*ms.begin() < K) {
        if (ms.size() == 1)
            return -1;
        auto it = ms.begin();
        int first = *it;
        ms.erase(it);
        ++it;
        int second = *it;
        ms.erase(it);
        ms.insert(first + second * 2);
        answer++;
    }
    return answer;
}

```
