#include <string>
#include <vector>
#include <set>

using namespace std;

int solution(vector<int> topping) {
    int answer = 0;
    set<int> a;
    vector<int> l;
    set<int> b;
    vector<int> r;
    int n = topping.size();
    for (int i = 0; i < n; i++) {
        a.insert(topping[i]);
        l.push_back(a.size());
    }
    for (int i = n-1; i >= 0; i--) {
        b.insert(topping[i]);
        r.push_back(b.size());
    }
    for (int i = 0; i < n-1; i++) {
        if (l[i] == r[n-2-i])
            answer++;
    }
    return answer;
}
