#include <string>
#include <vector>
#include <algorithm>

using namespace std;

int solution(vector<vector<int>> scores) {
    int answer = 0;
    int len = scores.size();
    vector<int> target = scores[0];
    int sum = target[0] + target[1];
    vector<int> scor;
    sort(scores.begin(), scores.end(), [](vector<int> &a, vector<int> &b){
        if (a[0] == b[0])
            return a[1] < b[1];
        return a[0] > b[0];
    });
    int pre = -1;
    for (int i = 0; i < len; i++) {
        if (scores[i][0] > target[0] && scores[i][1] > target[1])
            return -1;
        if (scores[i][1] >= pre) {
            scor.push_back(scores[i][0] + scores[i][1]);
            pre = scores[i][1];
        }
    }
    sort(scor.begin(), scor.end(), greater<int>());
    int r = 1;
    int cnt = 0;
    pre = -1;
    for (int i = 0; i < scor.size(); i++) {
        if (scor[i] != pre) {
            r += cnt;
            cnt = 0;
            pre = scor[i];
        }
        cnt++;
        if (scor[i] == sum)
            return r;
    }
    return -1;
}
