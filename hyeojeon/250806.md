#include <string>
#include <vector>
#include <map>
#include <algorithm>

using namespace std;

vector<int> getPrimes() {
    vector<bool> isPrime(100, true);
    vector<int> primes;
    isPrime[0] = false;
    isPrime[1] = false;
    
    for (int i = 2; i * i <= 100; ++i) {
        if (isPrime[i]) {
            for (int j = i * i; j <= 100; j += i)
                isPrime[j] = false;
        }
    }
    for (int i = 2; i <= 100; ++i) {
        if (isPrime[i])
            primes.push_back(i);
    }
    return primes;
}

map<int, int> primeFactorization(int n, const vector<int>& primes) {
    map<int, int> fac;
    for (int i = 0; i < primes.size(); i++) {
        if (primes[i] * primes[i] > n)
            break;
        while (n % primes[i] == 0) {
            fac[primes[i]]++;
            n /= primes[i];
        }
    }
    if (n > 1)
        fac[n]++;
    return fac;
}

int pow(int a, int b) {
    int res = 1;
    for (int i = 0; i < b; i++) {
        res *= a;
    }
    return res;
}

int solution(vector<int> arr) {
    int answer = 1;
    vector<int> primes = getPrimes();
    vector<map<int,int>> res;
    for (int i = 0; i < arr.size(); i++)
        res.push_back(primeFactorization(arr[i], primes));
    
    map<int, vector<int>> grouped;
    for (const auto& m : res) {
        for (const auto& [key, value] : m) {
            grouped[key].push_back(value);
        }
    }
    
    for (const auto& [key, vec] : grouped) {
        int tmp = *max_element(vec.begin(), vec.end());
        answer *= pow(key, tmp);
    }
    return answer;
}
