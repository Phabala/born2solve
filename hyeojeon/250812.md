struct Node {
    int val;
    Node *next;
};

class MyLinkedList {
    public:
        MyLinkedList() {
            head = NULL;
        }
        
        int get(int index) {
            int res = -1;
            Node *tmp = head;
            for (int i = 0; i < index; i++) {
                if (tmp->next == NULL)
                    return res;
                tmp = tmp->next;
            }
            if (tmp != NULL)
                res = tmp->val;
            return res;
        }
        
        void addAtHead(int val) {
            Node *newNode = new Node();
            newNode->val = val;
            newNode->next = NULL;
            if (head == NULL) {
                head = newNode;
            }
            else {
                newNode->next = head;
                head = newNode;
            }
        }
        
        void addAtTail(int val) {
            Node *tail = new Node();
            tail->val = val;
            tail->next = NULL;
            Node *tmp = head;
            if (head == NULL) {
                head = tail;
                return ;
            }
            while (tmp->next != NULL)
                tmp = tmp->next;
            tmp->next = tail;
        }
        
        void addAtIndex(int index, int val) {
            if (index == 0) {
                addAtHead(val);
                return ;
            }
            Node *n = new Node();
            n->val = val;
            n->next = NULL;
            Node *tmp = head;
            for (int i = 0; i < index - 1; i++) {
                if (tmp == NULL)
                    return ;
                tmp = tmp->next;
            }
            if (tmp == NULL)
                return ;
            Node *next = tmp->next;
            n->next = next;
            tmp->next = n;
        }
        
        void deleteAtIndex(int index) {
            Node *tmp = head;
            if (head == NULL)
                return ;
            if (index == 0) {
                Node *top = head;
                head = head->next;
                delete top;
                return ;
            }
            for (int i = 0; i < index - 1 && tmp->next != NULL; i++)
                tmp = tmp->next;
            if (tmp->next == NULL)
                return ;
            Node *del = tmp->next;
            tmp->next = del->next;
            delete del;
        }
    private:
        Node *head;
};
