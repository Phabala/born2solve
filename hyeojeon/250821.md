**코드(선택)**:

```cpp

#include <string>
#include <vector>
#include <algorithm>

using namespace std;

int solution(vector<int> citations) {
    int answer = 0;
    sort(citations.begin(), citations.end());
    int n = citations.size();
    int h = 0;
    for (int i = 0; i < n; i++) {
        if (n-i >= h) {
            h = (citations[i] > n-i ? n-i : citations[i]);
        }
    }
    return h;
}

```
