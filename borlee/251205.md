**오늘의 메모(선택)**: 
입출력 예 #2에 대해 고민이 많았는데..
malloc(0)도 malloc(1)도 테스트에 통과된다.;
아예 없다는 의미로 malloc(0)으로 제출했다.


**코드(선택)**
```c
#include <stdio.h>
#include <stdbool.h>
#include <stdlib.h>
#include <string.h>

// str_list_len은 배열 str_list의 길이입니다.
// 파라미터로 주어지는 문자열은 const로 주어집니다. 변경하려면 문자열을 복사해서 사용하세요.
char** solution(const char* str_list[], size_t str_list_len) {
    // return 값은 malloc 등 동적 할당을 사용해주세요. 할당 길이는 상황에 맞게 변경해주세요.
    size_t i;
    int idx = -1;
    char target = '\0';
    
    for (i = 0; i < str_list_len; i++)
    {
        char c = str_list[i][0];
        if (c == 'l' || c == 'r')
        {
            idx = (int)i;
            target = c;
            break;
        }
    }
    if (idx == -1)
        return malloc(0);
    if (target == 'l')
    {
        size_t left_count = idx;
        char **result = malloc(sizeof(char *) * left_count);
        if(!result)
            return NULL;
        for (i = 0; i < left_count; i++)
        {
            size_t len = strlen(str_list[i]);
            result[i] = malloc(len + 1);
            strcpy(result[i], str_list[i]);
        }
        return result;
    }
    else
    {
        size_t right_count = str_list_len - idx - 1;
        char **result = malloc(sizeof(char *) * right_count);
        if(!result)
            return NULL;
        for (i = 0; i < right_count; i++)
        {
            size_t len = strlen(str_list[idx + 1 + i]);
            result[i] = malloc(len + 1);
            strcpy(result[i], str_list[idx + 1 + i]);
        }
        return result;    
    }
}
```
