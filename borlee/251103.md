**코드(선택)**


```c
#include <stdio.h>
#include <stdbool.h>
#include <stdlib.h>

// 파라미터로 주어지는 문자열은 const로 주어집니다. 변경하려면 문자열을 복사해서 사용하세요.
char* solution(const char* my_string, int m, int c) {
    int len = 0;
    int i = 0;
    int row_count;
    char *answer;
    
    while (my_string[len])
        len++;
    row_count = len / m; // 단어의 갯수
    answer = malloc(row_count + 1);
    if (!answer)
        return NULL;
    while (i < row_count)
    {
        answer[i] = my_string[i * m + (c - 1)];
        i++;
    }
    answer[i] = '\0';
    return answer;
}
```
