**코드(선택)**


```c
#include <stdio.h>
#include <stdbool.h>
#include <stdlib.h>

int* solution(int n, int k) {
    // return 값은 malloc 등 동적 할당을 사용해주세요. 할당 길이는 상황에 맞게 변경해주세요.
    int count;
    int *answer;
    int i = 1;
    int j = 0;
    
    count = n / k ;
    answer = malloc(sizeof(int) *count);
    if (!answer)
        return NULL;
    while (i <= count)
    {
        answer[j] = k * i;
        i++;
        j++;
    }
    return answer;
    
}
```
