코딩 기초 트레이닝 
(대소문자 각각. 총 52개 문자의 갯수 세는 문제)

**코드(선택)**


```c
#include <stdio.h>
#include <stdbool.h>
#include <stdlib.h>

static bool is_upper(char c)
{
    return (c >= 'A' && c <= 'Z');
}

static bool is_lower(char c)
{
    return (c >= 'a' && c <= 'z');
}

// 파라미터로 주어지는 문자열은 const로 주어집니다. 변경하려면 문자열을 복사해서 사용하세요.
int* solution(const char* my_string) {
    // return 값은 malloc 등 동적 할당을 사용해주세요. 할당 길이는 상황에 맞게 변경해주세요.
    int* answer;
    int i = 0;
    unsigned char c;
    
    answer = calloc(52, sizeof(int));
    if(!answer)
        return NULL;
    while (my_string[i])
    {
        c = my_string[i];
        if (is_upper(c))
            answer[ c - 'A']++;
        else if (is_lower(c))
            answer[ c - 'a' + 26]++;
        i++;
    }
    return answer;
}
```
