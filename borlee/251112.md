**코드(선택)**
```c
#include <stdio.h>
#include <stdbool.h>
#include <stdlib.h>

// num_list_len은 배열 num_list의 길이입니다.
int solution(int num_list[], size_t num_list_len) 
{
    int idx = 0;
    while(idx < num_list_len)
    {
        if(num_list[idx] < 0)
            return idx;
        idx++;
    }
    return -1;
}
```

```c
#include <stdio.h>
#include <stdbool.h>
#include <stdlib.h>

// arr_len은 배열 arr의 길이입니다.
// intervals_rows는 2차원 배열 intervals의 행 길이, intervals_cols는 2차원 배열 intervals의 열 길이입니다.
int* solution(int arr[], size_t arr_len, int** intervals, size_t intervals_rows, size_t intervals_cols) {
    // return 값은 malloc 등 동적 할당을 사용해주세요. 할당 길이는 상황에 맞게 변경해주세요.
    int total_len = 0;
    int i, j;
    int start, end;
    
    for (i = 0; i < (int)intervals_rows; i++)
    {
        start = intervals[i][0];
        end = intervals[i][1];
        total_len += (end - start + 1);
    }
    int *result = malloc(sizeof(int) * total_len);
    if(!result)
        return NULL;
    
    int idx = 0;
    for (i = 0; i < (int)intervals_rows; i++)
    {
        start = intervals[i][0];
        end = intervals[i][1];
        for (j = start; j <= end; j++)
        {
            result[idx] = arr[j];
            idx++;
        }
    }
    return result;
}
```
