**오늘의 메모(선택)**: 
코딩 기초 트레이닝 
생각없이 first와 last를 0으로 초기화 했다가 욕을 봤다.


**코드(선택)**
```c
#include <stdio.h>
#include <stdbool.h>
#include <stdlib.h>

// arr_len은 배열 arr의 길이입니다.
int* solution(int arr[], size_t arr_len) {
    // return 값은 malloc 등 동적 할당을 사용해주세요. 할당 길이는 상황에 맞게 변경해주세요.
    int first = -1;
    int last = -1;
    size_t i;
 
    
    for(i = 0; i < arr_len; i++)
    {
        if (arr[i] == 2)
        {
            if (first == -1)
                first = (int)i;
            last = (int)i;
        }
    }
    if (first == -1) 
    {
        int *result = malloc(sizeof(int));
        if (!result)
            return NULL;
        result[0] = -1;
        return result;
    }
    int len = last - first + 1; 
    int *result = malloc(sizeof(int) * len);
    if (!result)
        return NULL;
    for (i = 0; i < (size_t)len; i++)
        result[i] = arr[first + i];
    return result;
}
```
