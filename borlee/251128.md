**오늘의 메모(선택)**: 
지난주에 풀었던 파이썬 코드를 다시 보고, 빼먹었던 타입힌트를 추가했다.
더불어 c언어로 바꾸기.

**코드(선택)**
```python
#include <stdio.h>
#include <stdbool.h>
#include <stdlib.h>

int explore(int current_energy, int cleared_count, int **dungeons, size_t dungeons_rows, bool *visited)
{
    int max_result = cleared_count;
    
    for (size_t i = 0; i < dungeons_rows; i++)
    {
        int required = dungeons[i][0];
        int cost = dungeons[i][1];
        
        if (current_energy >= required && visited[i] == false)
        {
            visited[i] = true;
            int result = explore(current_energy - cost, cleared_count + 1, dungeons, dungeons_rows, visited);
            if (result > max_result)
                max_result = result;
            visited[i] = false;
        }
    }
    return max_result;
}

// dungeons_rows는 2차원 배열 dungeons의 행 길이, dungeons_cols는 2차원 배열 dungeons의 열 길이입니다.
int solution(int k, int** dungeons, size_t dungeons_rows, size_t dungeons_cols) {
    bool *visited = malloc(sizeof(bool) * dungeons_rows);
    if(!visited)
        return -1;
    for (size_t i = 0; i < dungeons_rows; i++)
        visited[i] = false;
    int result = explore(k, 0, dungeons, dungeons_rows, visited);
    free(visited);
    return result;
}
```
