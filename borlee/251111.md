#include <stdio.h>
#include <stdbool.h>
#include <stdlib.h>

static int *slice_array(int *num_list, int start, int end, int step)
{
    int *result;
    int i = start;
    int j = 0;
    
    int count = ((end - start) / step) + 1;
    result = malloc(sizeof(int) * count);
    if(!result)
        return NULL;
    while (i <= end)
    {
        result[j] = num_list[i];
        i += step;
        j++;
    }
    return result;
}

// slicer_len은 배열 slicer의 길이입니다.
// num_list_len은 배열 num_list의 길이입니다.
int* solution(int n, int slicer[], size_t slicer_len, int num_list[], size_t num_list_len) {
    // return 값은 malloc 등 동적 할당을 사용해주세요. 할당 길이는 상황에 맞게 변경해주세요.
    int a = slicer[0];
    int b = slicer[1];
    int c = slicer[2];
    int start, end, step;
    
    if (n == 1)
    {
        start = 0;
        end = b;
        step = 1;
    }
    else if (n == 2)
    {
        start = a;
        end = (int)num_list_len - 1;
        step = 1;
    }
    else if (n == 3)
    {
        start = a;
        end = b;
        step = 1;
    }
    else
    {
        start = a;
        end = b;
        step = c;
    }
    return slice_array(num_list, start, end, step);
}
