**코드(선택)**
```c++
#include <string>
#include <vector>

using namespace std;

bool isValid(const vector<int>& colPositions, int currentRow, int targetCol)
{
    for (int prevRow = 0; prevRow < currentRow; ++prevRow)
    {
        if (colPositions[prevRow] == targetCol)
            return false;
        if (abs(colPositions[prevRow] - targetCol) == (currentRow - prevRow))
            return false;
    }
    return true;
}

int backtrack(vector<int>& colPositions, int currentRow, int n)
{
    if(currentRow == n)
        return 1;
    int totalSol = 0;
    for(int col = 0; col < n; ++col)
    {
        if (isValid(colPositions, currentRow, col))
        {
            colPositions[currentRow] = col;
            totalSol += backtrack(colPositions, currentRow + 1, n);
        }
    }
    return totalSol;
}

int solution(int n) 
{
    vector<int> colPositions(n, 0);
    return backtrack(colPositions, 0, n);
}
```
