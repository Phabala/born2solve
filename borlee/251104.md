#include <stdio.h>

#include <stdlib.h>

// 파라미터로 주어지는 문자열은 const로 주어집니다. 변경하려면 문자열을 복사해서 사용하세요.
char* solution(int q, int r, const char* code) {
    // return 값은 malloc 등 동적 할당을 사용해주세요. 할당 길이는 상황에 맞게 변경해주세요.
    int i = 0;
    int j = 0;
    int len = 0;
    char *answer;
    
    while (code[len])
        len++;
    answer = malloc(len + 1);
    if(!answer)
        return NULL;
    while (i < len)
    {
        if (i % q == r)
        {
            answer[j] = code[i];
            j++;
        }
        i++;
    }
    answer[j] = '\0';
    return answer;
}
