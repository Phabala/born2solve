int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int n, m, k;

    std::cin >> n >> m >> k;

    if (k < n + m - 1) {
        std::cout << "NO\n";
        return 0;
    }

    std::cout << "YES\n";
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++)
            std::cout << i + j + 1 << ' ';
        std::cout << '\n';
    }
    return 0;
}
