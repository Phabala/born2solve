#include <algorithm>

#include <string>
#include <unordered_set>
#include <utility>
#include <vector>

int main() {
    int N;
    std::cin >> N;
    if (N & 1) {
        int mid = (N + 1) / 2;
        std::cout << mid;
        for (int i = N; i > mid; i--)
            std::cout << ' ' << i << ' ' << i - mid;
    } else {
        int mid = N / 2;
        for (int i = N; i > mid; i--)
            std::cout << i - mid << ' ' << i << ' ';
    }
    return 0;
}
