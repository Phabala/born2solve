#define TAEBKIM                                                                \
    std::ios::sync_with_stdio(false);                                          \
    std::cin.tie(NULL);                                                        \
    std::cout.tie(NULL);
;

using namespace std;

int main() {
    TAEBKIM;
    int t;
    cin >> t;
    string s, p;

    while (t--) {
        int ans = 0;
        cin >> s >> p;

        int start = 0;

        while (start < s.size()) {
            if (s.substr(start, p.size()) == p)
                start += p.size();
            else
                start++;
            ans++;
        }
        cout << ans << "\n";
    }

    return 0;
}
