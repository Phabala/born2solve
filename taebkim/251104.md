using namespace std;

int main() {
    TAEBKIM;

    int n;
    cin >> n;

    int a[101];

    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }

    int b[101];

    for (int i = 0; i < n; i++) {
        cin >> b[i];
    }

    int answer = 0;
    for (int i = 0; i < n; i++) {
        int diff = b[i] - a[i];
        if (diff > 0) {
            answer += diff;
        }
    }

    cout << answer;

    return 0;
}
