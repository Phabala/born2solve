#define TAEBKIM                                                                \
    std::ios::sync_with_stdio(false);                                          \
    std::cin.tie(NULL);                                                        \
    std::cout.tie(NULL);
;

using namespace std;

int main() {
    TAEBKIM;

    int N;

    cin >> N;

    vector<int> v(N, 0);
    vector<int> fruits(10, 0);

    for (int i = 0; i < N; i++) {
        cin >> v[i];
    }

    int l_ptr = 0, r_ptr = 0;

    int result = 0;

    while (r_ptr < N) {
        fruits[v[r_ptr++]]++;

        while(10 - count(fruits.begin(), fruits.end(), 0) > 2)
            fruits[v[l_ptr++]]--;
        result = max(result, r_ptr - l_ptr);
    }

    cout << result << "\n";

    return 0;
}
