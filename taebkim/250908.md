        int a, b, c;
        std::cin >> a >> b >> c;

        if (a == b - c) {
            std::cout << "does not matter" << "\n";
        } else if (a < b - c) {
            std::cout << "advertise" << "\n";
        } else {
            std::cout << "do not advertise" << "\n";
        }
    }

    return 0;
}
