**코드(선택)**:

```c++
#include <algorithm>
#include <cmath>
#include <iomanip>
#include <iostream>
#include <map>
#include <numeric>
#include <queue>
#include <set>
#include <stack>
#include <stdexcept>
#include <string>
#include <unordered_set>
#include <utility>
#include <vector>

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int t;
    std::cin >> t;

    int n;
    std::string s;
    for (int i = 0; i < t; i++) {
        std::cin >> n;
        std::cin >> s;

        int idx = 0;

        while (s[idx]) {
            std::string abb = s.substr(idx, 3);

            if (abb == "ABB") {
                s[idx] = 'B';
                s[idx + 1] = 'A';
                s.erase(idx + 2, 1);

                idx = 0;
            } else {
                idx++;
            }
        }

        std::cout << s << "\n";
    }

    return 0;
}
```
