#include <algorithm>

#include <stack>

#include <string>
#include <unordered_set>
#include <utility>
#include <vector>

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int N, M;
    std::vector<std::pair<std::pair<int, int>, char>> vec;

    std::cin >> N >> M;
    while (M--) {
        int i, j;
        char c;
        std::cin >> i >> j >> c;
        vec.emplace_back(std::make_pair(std::make_pair(j, i), c));
    }
    sort(vec.begin(), vec.end());
    for (auto &p : vec)
        std::cout << p.second;

    return 0;
}
