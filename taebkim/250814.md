int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int n;
    std::cin >> n;

    char s;
    int sum = 0;

    for (int i = 0; i < n; i++) {
        std::cin >> s;

        if (s == 'O')
            sum++;
    }

    if ((n / 2 + n % 2) <= sum)
        std::cout << "Yes" << "\n";
    else
        std::cout << "No" << "\n";

    return 0;
}
