#include <algorithm>
#include <cmath>
#include <iomanip>
#include <iostream>
#include <map>

#include <queue>
#include <set>
#include <stack>
#include <stdexcept>
#include <string>
#include <unordered_set>
#include <utility>
#include <vector>

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int count = 1;
    while (1) {
        double x;

        std::cin >> x;
        if (x == 0)
            break;
        double y, z;
        std::cin >> y >> z;
        if (x >= sqrt(z / 2 * z / 2 + y / 2 * y / 2)) {
            std::cout << "Pizza " << count << " fits on the table.\n";
        } else {
            std::cout << "Pizza " << count << " does not fit on the table.\n";
        }
        count++;
    }

    return 0;
}
