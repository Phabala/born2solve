**코드(선택)**:

```c++
#include <algorithm>
#include <cmath>
#include <iomanip>
#include <iostream>
#include <map>
#include <numeric>
#include <queue>
#include <set>
#include <stack>
#include <stdexcept>
#include <string>
#include <unordered_set>
#include <utility>
#include <vector>

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    long long result = 0;
    long long a, b;

    std::cin >> a >> b;

    long long first = a;
    if (first < 4) {
        first = 4;
    }
    if (first % 2 != 0) {
        first++;
    }

    long long last = b;
    if (last % 2 != 0) {
        last--;
    }

    if (first <= last) {
        long long start_term = first / 2;
        long long end_term = last / 2;
        long long count = end_term - start_term + 1;

        result = count * (start_term + end_term);
    }

    std::cout << result << "\n";

    return 0;
}
```
