**코드(선택)**:

```c++
#include <algorithm>
#include <cmath>
#include <iomanip>
#include <iostream>
#include <map>
#include <numeric>
#include <queue>
#include <set>
#include <stack>
#include <stdexcept>
#include <string>
#include <unordered_set>
#include <utility>
#include <vector>

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    std::unordered_set<int> _s;
    int n;
    std::cin >> n;

    int ans = 0;
    for (int i = 0; i < n; i++) {
        int a, b;
        std::cin >> a >> b;
        if (b) {
            if (_s.find(a) != _s.end())
                ans++;
            else
                _s.insert(a);
        } else {
            if (_s.find(a) == _s.end())
                ans++;
            else
                _s.erase(a);
        }
    }
    std::cout << ans + _s.size();

    return 0;
}
```
