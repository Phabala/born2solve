```c++
#include <algorithm>
#include <cmath>
#include <iomanip>
#include <iostream>
#include <map>
#include <numeric>
#include <queue>
#include <set>
#include <stack>
#include <stdexcept>
#include <string>
#include <unordered_set>
#include <utility>
#include <vector>

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int n;
    std::cin >> n;
    std::cin.ignore();

    for (int i = 1; i <= n; i++) {
        std::string str;
        std::getline(std::cin, str);

        transform(str.begin(), str.end(), str.begin(), ::tolower);

        int arr[26] = {0};
        int cnt = 0;
        int len = str.length();

        for (int i = 0; i < len; i++) {
            cnt = str[i] - 97;
            arr[cnt]++;
        }

        int cnt1 = 0;
        int cnt2 = 0;
        int cnt3 = 0;

        for (int i = 0; i < 26; i++) {
            if (arr[i] != 0)
                cnt1++;
            if (arr[i] == 2)
                cnt2++;
            if (arr[i] >= 3)
                cnt3++;
        }

        if (cnt3 == 26) {
            std::cout << "Case " << i << ": " << "Triple pangram!!!\n";
        } else if (cnt2 == 26 || cnt2 + cnt3 == 26) {
            std::cout << "Case " << i << ": " << "Double pangram!!\n";
        } else if (cnt1 == 26) {
            std::cout << "Case " << i << ": " << "Pangram!\n";
        } else {
            std::cout << "Case " << i << ": " << "Not a pangram\n";
        }
    }

    return 0;
}
```
