using namespace std;

long long dp[31][31];

void combi(int n, int r) {
    for (int i = 1; i <= r; i++) {
        dp[1][i] = i;
    }

    for (int i = 2; i <= n; i++) {
        for (int j = i; j <= r; j++) {
            dp[i][j] = dp[i][j - 1] + dp[i - 1][j - 1];
        }
    }
}

int main() {
    TAEBKIM;

    int t;
    int n, m;
    cin >> t;

    while (t--) {
        cin >> n >> m;
        combi(n, m);
        cout << dp[n][m] << '\n';
    }

    return 0;
}
