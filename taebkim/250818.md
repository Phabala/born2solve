```c++
#include <algorithm>
#include <cmath>
#include <iomanip>
#include <iostream>
#include <map>
#include <numeric>
#include <queue>
#include <set>
#include <stack>
#include <stdexcept>
#include <string>
#include <unordered_set>
#include <utility>
#include <vector>

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int n, k;
    std::cin >> n >> k;
    std::cin.ignore();
    std::string st;
    std::cin >> st;

    std::vector<char> s;
    for (int i = st.length() - 1; i >= 0; i--) {
        if (s.empty() && !(st[i] == 'A' || st[i] == 'O' || st[i] == 'I' ||
                           st[i] == 'U' || st[i] == 'E')) {
            s.push_back(st[i]);
        } else if ((s.size() == 1 || s.size() == 2) && st[i] == 'A') {
            s.push_back(st[i]);
        } else if (s.size() >= 3) {
            s.push_back(st[i]);
        }

        if (s.size() == k) {
            break;
        }
    }

    if (s.size() != k) {
        std::cout << "NO" << "\n";
    } else {
        std::cout << "YES" << "\n";
        for (int i = k - 1; i >= 0; i--) {
            std::cout << s[i];
        }
        std::cout << "\n";
    }

    return 0;
}
```
