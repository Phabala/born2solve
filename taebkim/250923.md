#include <algorithm>
#include <cmath>
#include <iomanip>
#include <iostream>
#include <map>

#include <set>
#include <stack>
#include <stdexcept>
#include <string>
#include <unordered_set>
#include <utility>
#include <vector>

#define TAEBKIM                                                                \
    std::ios::sync_with_stdio(false);                                          \
    std::cin.tie(NULL);                                                        \
    std::cout.tie(NULL);

int main() {
    TAEBKIM;

    int k, n;
    std::cin >> k >> n;

    long long cable[10001];
    long long max_cable = 0;
    for (int i = 0; i < k; i++) {
        std::cin >> cable[i];
        max_cable = std::max(max_cable, cable[i]);
    }

    long long left = 1;
    long long right = max_cable;
    long long mid;

    long long answer = 0;
    while (left <= right) {
        long long result = 0;
        mid = (left + right) / 2;
        for (int i = 0; i < k; i++) {
            result += cable[i] / mid;
        }

        if (result >= n) {
            left = mid + 1;
            answer = std::max(answer, mid);
        } else {
            right = mid - 1;
        }
    }

    std::cout << answer;

    return 0;
}
