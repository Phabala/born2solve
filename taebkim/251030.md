using namespace std;

string convert(long long int num, int n) {
    if (num == 0) {
        return "0";
    }
    string result;

    while (num > 0) {
        int remainder = num % n;
        if (remainder < 10) {
            result += to_string(remainder);
        } else {
            result += (char)('A' + remainder - 10);
        }
        num /= n;
    }
    reverse(result.begin(), result.end());
    return result;
}

int main() {
    TAEBKIM;

    int T;
    long long int A;
    int n;

    cin >> T;
    for (int i = 0; i < T; i++) {
        cin >> A >> n;

        string palindrome = convert(A, n);
        string back = palindrome;
        reverse(back.begin(), back.end());

        if (palindrome == back)
            cout << "1" << "\n";
        else
            cout << "0" << "\n";
    }

    return 0;
}
