**오늘의 메모(선택)**:

**코드(선택)**:
```cpp
#include <map>
#include <iostream>

int	main()
{
	std::string fruit;
	int num;
	std::map<std::string, int> fruits;

	int n;
	std::cin >> n;
	for (int i = 0; i < n; ++i)
	{
		std::cin >> fruit >> num;
		fruits[fruit] += num;
	}

	std::map<std::string, int>::iterator iter;
	for (iter = fruits.begin(); iter != fruits.end(); ++iter)
	{
		if (iter->second == 5)
		{
			std::cout << "YES";
			return 0;
		}
	}
	std::cout << "NO";

	return 0;
}
```
