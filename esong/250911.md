**오늘의 메모(선택)**:

**코드(선택)**:
```cpp
#include <iostream>
#include <sstream>

int main()
{
	std::string str, temp;
	int left_sum, right_sum, current_nbr;
	bool found_minus = false;
	std::stringstream ss;

	std::cin >> str;

	for (int i = 0; i < str.size(); ++i)
	{
		if (str[i] == '+' || str[i] == '-')
		{	
			ss.str(temp);
			ss >> current_nbr;

			if (found_minus == true)
				right_sum += current_nbr;
			else
				left_sum += current_nbr;
			
			if (str[i] == '-')
				found_minus = true;

			temp.clear();
			ss.clear();
		}
		else
		{
			temp += str[i];
		}
	}

	ss.str(temp);
	ss >> current_nbr;
	if (found_minus == true)
		right_sum += current_nbr;
	else
		left_sum += current_nbr;

	std::cout << left_sum - right_sum;

	return 0;
}
```
