**코드(선택)**:
```cpp
#include <iostream>
#include <deque>

int	main()
{
	int	cardsCnt;
	std::cin >> cardsCnt;

	std::deque<int>	cards;
	for (int i = 0; i < cardsCnt; ++i)
		cards.push_back(i + 1);
	
	int	tempCard;
	for (int i = 0; i < cardsCnt - 1; ++i)
	{
		std::cout << cards.front() << " ";
		cards.pop_front();
		
		tempCard = cards.front();
		cards.pop_front();
		cards.push_back(tempCard);
	}
	std::cout << cards.front();
	
	return 0;
}
```
