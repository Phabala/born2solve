**오늘의 메모(선택)**:

**코드(선택)**:
```cpp
#include <iostream>
#include <unordered_set>

int	main()
{
	std::ios::sync_with_stdio(0);
	std::cin.tie(0);

	int	testCaseCount, n, m;
	std::unordered_set<int>	note;

	std::cin >> testCaseCount;
	int	nbr;
	while (testCaseCount--)
	{
		std::cin >> n;
		while (n--)
		{
			std::cin >> nbr;
			note.insert(nbr);
		}

		std::cin >> m;
		while (m--)
		{
			std::cin >> nbr;
			if (note.count(nbr) > 0)
				std::cout << "1\n";
			else
				std::cout << "0\n";
		}
		note.clear();
	}

	return 0;
}
```
